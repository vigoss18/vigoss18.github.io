题意是问你加上几条边之后，能把整个图变成一个强连通分量。

做法其实也很简单，你先做一遍tarjan，然后把所有属于同一个强连通分量的点缩成一个点之后，然后只要遍历这些点，max(出度为0的点,入度为0的点)就是答案，你画个图就知道了。

刚开始写的时候，我发现我连强连通分量都判断错误了，仔细一看，原来是自己的模板有问题，好，改模板，好不容易改对了，交，wa，再看，啊，栈没有清空，再交，wa，再看，看不出个所以然了，直到现在才突然发现，自己遍历缩好的点的时候其实还是在遍历全部的点，QAQ我只好大呼一声“我真是傻逼啊”，于是就ac了0.0.也不能说自己这几个小时的debug毫无意义吧，至少我更加熟悉了对tarjan模板的应用，不过tarjan还有用来求lca啥的别的模板，等ec回来还要多加学习啊。

#include <bits/stdc++.h>
using namespace std;
#define MAXN 20050
#define MAXV 200010

struct Edge
{
    int to,next,w;
}edge[MAXV];
int tot,head[MAXN];

void add(int s,int u,int w)
{
    edge[tot].to = u;
    edge[tot].w = w;
    edge[tot].next = head[s];
    head[s] = tot++;
}

int n,m;
int dfn[MAXN],low[MAXN],col[MAXN];
stack <int>sta;
bool insta[MAXN];
int index,sum,scc,top;

void tarjan(int u)
{
    dfn[u] = low[u] = ++index;
    sta.push(u);
    insta[u] = true;
    for(int i = head[u];i;i = edge[i].next)
    {
        int v = edge[i].to;
        if(!dfn[v])
        {
            tarjan(v);
            low[u] = min(low[u],low[v]);
        }
        else if(insta[v])
        {
            low[u] = min(low[u],dfn[v]);
        }
    }
    if(dfn[u] == low[u])
    {
        //cout << scc << " " << u << endl;
        scc++;
        int x;
        do{
            x = sta.top();sta.pop();
            //cout << x << " " << scc << endl;
            insta[x] = false;
            col[x] = scc;
        }while(x != u);
    }
}

int main(){
    int t;
    cin >> t;
    while(t--)
    {
        cin >> n >> m;
        int a,b;
        memset(dfn,0,sizeof(dfn));
        memset(col,0,sizeof(col));
        memset(head,0,sizeof(head));
        memset(insta,0,sizeof(insta));
        while(!sta.empty()) sta.pop();
        tot = 1;
        index = 0;scc = 0;
        int sv1[50050],sv2[50050];
        for(int i = 0;i < m;i++)
        {
            scanf("%d %d",&a,&b);
            sv1[i] = a,sv2[i] = b;
            add(a,b,1);
        }
        for(int i = 1;i <= n;i++)
            if(!dfn[i])
                tarjan(i);
        //for(int i = 1;i <= n;i++)
        //    cout << col[i];
        if(scc == 1)
        {
            cout << 0 << endl;
            continue;
        }
        int f1[MAXN],f2[MAXN];
        for(int i = 1;i <= scc;i++)
            f1[i] = 0,f2[i] = 0;
        for(int i = 0;i < m;i++)
        {
            if(col[sv1[i]] != col[sv2[i]])
                f1[col[sv1[i]]]++,f2[col[sv2[i]]]++;
        }
        int ma = 0,mb = 0;
        for(int i = 1;i <= scc;i++)
        {
            if(!f1[i])
                ma++;
            if(!f2[i])
                mb++;
            //cout << ma << " " << mb << endl;
        }
        cout << max(ma,mb) << endl;
    }
    return 0;
}
