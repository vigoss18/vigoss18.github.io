##E·Repeated Substrings
听浙大的学长说是一道后缀数组的题目。题意是给你一串字符串，找出其中有哪些子串重复出现过，统计其数量。按第一个样例来解释

aabaab后缀按照字典序来排就是

aab
aabaab
ab
abaab
b
baab
那么height数组就是0,3,1,2,0,1

在每一段升序中，没有出现过的子串就是要找的子串，求和即可。这里我找了一个网上的后缀数组模板用。

#include<bits/stdc++.h>
#define LL long long
#define MOD 1000000007
#define PI acos(-1.0)
#define N 100005
using namespace std;

int wa[N],wb[N],wsf[N],wv[N],sa[N];
int rank1[N],height[N],s[N];
//sa:字典序中排第i位的起始位置在str中第sa[i]
//rank:就是str第i个位置的后缀是在字典序排第几
//height：字典序排i和i-1的后缀的最长公共前缀
int cmp(int *r,int a,int b,int k)
{
    return r[a]==r[b]&&r[a+k]==r[b+k];
}
void getsa(int *r,int *sa,int n,int m)//n要包含末尾添加的0
{
    int i,j,p,*x=wa,*y=wb,*t;
    for(i=0; i<m; i++)  wsf[i]=0;
    for(i=0; i<n; i++)  wsf[x[i]=r[i]]++;
    for(i=1; i<m; i++)  wsf[i]+=wsf[i-1];
    for(i=n-1; i>=0; i--)  sa[--wsf[x[i]]]=i;
    p=1;
    j=1;
    for(; p<n; j*=2,m=p)
    {
        for(p=0,i=n-j; i<n; i++)  y[p++]=i;
        for(i=0; i<n; i++)  if(sa[i]>=j)  y[p++]=sa[i]-j;
        for(i=0; i<n; i++)  wv[i]=x[y[i]];
        for(i=0; i<m; i++)  wsf[i]=0;
        for(i=0; i<n; i++)  wsf[wv[i]]++;
        for(i=1; i<m; i++)  wsf[i]+=wsf[i-1];
        for(i=n-1; i>=0; i--)  sa[--wsf[wv[i]]]=y[i];
        t=x;
        x=y;
        y=t;
        x[sa[0]]=0;
        for(p=1,i=1; i<n; i++)
            x[sa[i]]=cmp(y,sa[i-1],sa[i],j)? p-1:p++;
    }
}
void getheight(int *r,int n)//n不保存最后的0
{
    int i,j,k=0;
    for(i=1; i<=n; i++)  rank1[sa[i]]=i;
    for(i=0; i<n; i++)
    {
        if(k)
            k--;
        else
            k=0;
        j=sa[rank1[i]-1];
        while(r[i+k]==r[j+k])
            k++;
        height[rank1[i]]=k;
    }
}

char str[N];

int main()
{
    int t;
    cin >> t;
    getchar();
    while(t--)
    {
        gets(str);
        int len = strlen(str);
        for(int i = 0;i < len;i++)
            s[i] = str[i];
        s[len] = 0;
        getsa(s,sa,len + 1,200);
        getheight(s,len);
        int maxn = 0,minn = 0,sum = 0;
        for(int i = 1;i <= len;i++)
        {
            if(height[i] > maxn)
            {
                maxn = height[i];
            }
            else
            {
                sum += maxn - minn;
                minn = height[i];
                maxn = height[i];
            }
        }
        sum += maxn - minn;
        cout << sum << endl;
    }
return 0;
}

##F·Landline Telephone Network
题意就是要构成一棵树，由安全节点和不安全节点组成，所有的不安全节点都只能是子节点，求最小费用。先用并查集把安全节点搞成一棵树，然后再把不安全节点连上去就行。
