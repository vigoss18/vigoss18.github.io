最大匹配解法学习

发表于 2015-11-10 | 分类于 ACM |

这两天抽空看了一下最大匹配的解法，发现还真的蛮常用的，学习，学习一下。不得不说kuangbin的blog真的很有用啊，不愧是大神……学习地址1学习地址2

所谓最大匹配，最简单的理解方法就是字面意思了，最大，匹配，就是给你一个图，让你找出一种选择配对的情况使得最多的点能互相两两配对。那么如何来求得最大配对呢？就要引入几个定义了。

交替路：从一个未匹配点出发，依次经过非匹配边、匹配边、非匹配边…形成的路径叫交替路。

增广路：从一个未匹配点出发，走交替路，如果途径了一个未匹配点，那么这条路就是增光路。此时匹配边比未匹配边少一条，只要互换就可以改进匹配了。

学完这些概念，我脑子里第一个蹦出来的是用dfs来写，不过感觉写起来有点困难，就copy了一份别人的模板来用，确实很短啊。

//************************************************

int linker[MAXN];
bool used[MAXN];
vector<int>mp[MAXN];
int uN;
int dirx[] = {0,1,0,-1},diry[] = {1,0,-1,0};
bool dfs(int u)
{
    for(int i=0;i<mp[u].size();i++)
    {
        if(!used[mp[u][i]])
        {
            used[mp[u][i]]=true;
            if(linker[mp[u][i]]==-1||dfs(linker[mp[u][i]]))
            {
                linker[mp[u][i]]=u;
                return true;
            }
        }
    }
    return false;
}
int hungary()
{
    int u;
    int res=0;
    memset(linker,-1,sizeof(linker));
    for(u=1;u<=uN;u++)
    {
        memset(used,false,sizeof(used));
        if(dfs(u)) res++;
    }
    return res;
}

bfs的版本在上面的链接里也有，比较长，但是好理解一点，两者在实际应用中的区别是，dfs版本简短，但是效率低，适用于稀疏图，bfs版本适用于稠密图。

先补充一下定义：

最大匹配数：最大匹配的匹配边的数目

最小点覆盖数：选取最少的点，使任意一条边至少有一个端点被选择

独立集：图中任意两个顶点都不相连的顶点集合。

最小路径覆盖数：对于一个 DAG（有向无环图），选取最少条路径，使得每个顶点属于且仅属于一条路径。路径长可以为 0（即单个点）。

那么我就可以先映入三个定理了。

定理1：最大匹配数 = 最小点覆盖数（这是 Konig 定理）

定理2：最大匹配数 = 顶点数 - 最大独立集

定理3：最小路径覆盖数 = 顶点数 - 最大匹配数

一定要牢记这三个定理，不然到时候要求什么都不知道啊0.0

int linker[MAXN];

int main()
{
    int n,m,mapp[105][105];
    while(cin >> n >> m && n + m)
    {
        uN = 0;
        for(int i = 0;i < MAXN;i++)
            mp[i].clear();
        memset(mapp,-1,sizeof(mapp));
        int c;
        cin >> c;
        for(int i = 0;i < c;i++)
        {
            int r,b;
            cin >> r >> b;
            mapp[r][b] = 0;
        }
        for(int i = 1;i <= n;i++)
        {
            for(int j = 1;j <= m;j++)
            {
                if(mapp[i][j] == -1)
                {
                    mapp[i][j] = ++uN;
                }
            }
        }
        for(int i = 1;i <= n;i++)
        {
            for(int j = 1;j <= m;j++)
            {
                if(mapp[i][j])
                {
                    for(int k = 0;k < 4;k++)
                    {
                        int rx = i + dirx[k],ry = j + diry[k];
                        if(rx >= 1 && rx <= n && ry >= 1 && ry <= m && mapp[rx][ry])
                        {
                            mp[mapp[i][j]].push_back(mapp[rx][ry]);
                            mp[mapp[rx][ry]].push_back(mapp[i][j]);
                        }
                    }
                }
            }
        }
        cout << hungary() / 2 << endl;
    }
}

题意是给你一个图，有空地，墙,草地，要求在同一条线上除非有墙隔着，不然就不能放多个机器人（机器人必须在空地上），应该是缩点后进行最大匹配吧。

终于过啦，一开始的时候我把每一个空地都当作一个点来做，结果算出来不对，我特“聪明”地发现比真实答案都小了1，就直接在答案后面+1，然而这并没有什么卵用。然后我偷偷地瞄了一眼题解，噫，真的就一眼噢，发现应该把墙一边的点全部缩成一块。于是我的做法就是，先在行上进行缩块，然后在遍历列的时候，边把列缩块，边把匹配的块标记起来，然后套一下模板就过了，真开心啊~

#define ll long long
#define MAXN 100005
using namespace std;

int linker[MAXN];
bool used[MAXN];
vector<int>mp[MAXN];
int uN;
int dirx[] = {0,1,0,-1},diry[] = {1,0,-1,0};
bool dfs(int u)
{
    for(int i=0;i<mp[u].size();i++)
    {
        if(!used[mp[u][i]])
        {
            used[mp[u][i]]=true;
            if(linker[mp[u][i]]==-1||dfs(linker[mp[u][i]]))
            {
                linker[mp[u][i]]=u;
                return true;
            }
        }
    }
    return false;
}
int hungary()
{
    int u;
    int res=0;
    memset(linker,-1,sizeof(linker));
    for(u=1;u<=uN;u++)
    {
        memset(used,false,sizeof(used));
        if(dfs(u)) res++;
    }
    return res;
}

int main()
{
    int t,cas = 1;
    cin >> t;
    while(t--)
    {
        for(int i = 0;i < MAXN;i++)
            mp[i].clear();
        int n,m,mapp[55][55],p[55][55],pp[55][55];
        uN = 0;
        memset(mapp,0,sizeof(mapp));
        memset(p,0,sizeof(p));
        memset(pp,0,sizeof(pp));
        string s[55];
        cin >> n >> m;
        for(int i = 0;i < n;i++)
        {
            cin >> s[i];
        }
        for(int i = 0;i < n;i++)        //对行进行分块
        {
            for(int j = 0;j < m;j++)
            {
                if(s[i][j] == 'o' && !p[i][j])
                {
                    p[i][j] = ++uN;
                    for(int k = j + 1;k < m && s[i][k] != '#';k++)
                    {
                        if(s[i][k] == 'o')
                            p[i][k] = uN;
                    }
                }
            }
        }
        for(int i = 0;i < m;i++)         //边分块，边标记匹配块
        {
            for(int j = 0;j < n;j++)
            {
                if(s[j][i] == 'o' && !pp[j][i])
                {
                    pp[j][i] = 1;
                    uN++;
                    mp[uN].push_back(p[j][i]);
                    mp[p[j][i]].push_back(uN);
                    for(int k = j + 1;k < n && s[k][i] != '#';k++)
                    {
                        if(s[k][i] == 'o')
                        {
                            pp[k][i] = 1;
                            mp[uN].push_back(p[k][i]);
                            mp[p[k][i]].push_back(uN);
                        }
                    }
                }
            }
        }
        printf("Case :%d\n%d\n",cas++,hungary() / 2);
    }
}

这道题其实也很简单，只是一开始建图建错了，题意是有n个人，给出每个人喜欢的动物名和不喜欢的动物名，然后让你去掉一些动物之后求出最多还有几个人不会走（存在喜欢的动物且不存在不喜欢的动物），我一开始把猫和狗用来建图了，但其实应该是吧人的关系建图，两个人的喜恶关系交叉的时候要建立一条线。就可以了。

#include <bits/stdc++.h>
#define ll long long
#define MAXN 100500
using namespace std;

int linker[MAXN];
bool used[MAXN];
vector<int>mp[MAXN];
int uN;
int dirx[] = {0,1,0,-1},diry[] = {1,0,-1,0};
bool dfs(int u)
{
    for(int i=0;i<mp[u].size();i++)
    {
        if(!used[mp[u][i]])
        {
            used[mp[u][i]]=true;
            if(linker[mp[u][i]]==-1||dfs(linker[mp[u][i]]))
            {
                linker[mp[u][i]]=u;
                return true;
            }
        }
    }
    return false;
}
int hungary()
{
    int u;
    int res=0;
    memset(linker,-1,sizeof(linker));
    for(u=0;u<uN;u++)
    {
        memset(used,false,sizeof(used));
        if(dfs(u)) res++;
    }
    return res;
}

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        for(int i = 0;i < MAXN;i++)
            mp[i].clear();
       int n,m,k;
       pair<string,string>p[MAXN];
       cin >> n >> m >> k;
       uN = k;
       for(int i = 0;i < k;i++)
       {
            cin >> p[i].first >> p[i].second;
       }
       for(int i = 0;i < k;i++)
       {
            for(int j = 0;j < k;j++)
            {
                if(p[i].first == p[j].second || p[i].second == p[j].first)
                    mp[i].push_back(j),mp[j].push_back(i);
            }
       }
       cout << k - hungary() / 2 << endl;
    }
}
