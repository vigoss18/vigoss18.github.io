2014年欧洲区域赛的一道题，小学奥数吧，就是给你代表数字的字母，让你找找有几种匹配的方式。我的bug是没有判断第一个数字是不是0，因为不能出现前导0。其他也没什么了。

#include<bits/stdc++.h>
typedef long long LL;
using namespace std;

int n,ans,vis[10];
string s[10];
map<char,int>p;

bool check()
{
    for(int i = 0;i < n;i++)
    {
        for(int j = 0;j < s[i].size();j++)
        {
            if(s[i][j] != '?')
            {
                if(p[s[i][j]] == 0)
                    return false;
                break;
            }
        }
    }
    return true;
}

void dfs(int pos,int dep,int pl)
{

    if(pos == -1)
    {
        if(pl == 0 && check())
        {
            ans++;
        }

        return;
    }
    if(dep == n - 1)
    {
        int sum = pl;
        for(int i = 0;i < n - 1;i++)
        {
            if(p[s[i][pos]] == '?')
                continue;
            sum += p[s[i][pos]];
        }
        if(p[s[dep][pos]] == -1)
        {
            if(vis[sum % 10])
                return;
            p[s[dep][pos]] = sum % 10;
            vis[sum % 10] = 1;
            pl = sum / 10;
            dfs(pos - 1,0,pl);
            vis[sum % 10] = 0;
            p[s[dep][pos]] = -1;
        }
        else if(p[s[dep][pos]] != sum % 10)
        {
            return;
        }
        else
        {
            pl = sum / 10;
            dfs(pos - 1,0,pl);
        }
    }
    else
    {
        if(s[dep][pos] == '?')
            dfs(pos,dep + 1,pl);
        else
        {
            if(p[s[dep][pos]] != -1)
            {
                dfs(pos,dep + 1,pl);
            }
            else
            {
                for(int i = 0;i < 10;i++)
                {
                    if(!vis[i])
                    {
                        p[s[dep][pos]] = i;
                        vis[i] = 1;
                        dfs(pos,dep + 1,pl);
                        vis[i] = 0;
                        p[s[dep][pos]] = -1;
                    }
                }
            }
        }

    }
    return;
}

int main()
{
    //freopen("1.txt","r",stdin);
    while(cin >> n)
    {
        for(int i = 0;i < 26;i++)
            p['A' + i] = -1;
        memset(vis,0,sizeof(vis));
        int ma = 0;
        for(int i = 0;i < n;i++)
        {
            cin >> s[i];
            ma = max(ma,(int)s[i].size());
        }
        if(s[n - 1].size() < ma)
            cout << 0 << endl;
        else
        {
            for(int i = 0;i < n;i++)
            {
                while(s[i].size() < ma)
                {
                    s[i].insert(0,"?");
                }
            }
            ans = 0;
            dfs(s[0].size() - 1,0,0);
            cout << ans << endl;
        }
    }
    return 0;
}

#include<bits/stdc++.h>
#define ll long long
#define MAXN 6333333
#define INF 9999999
using namespace std;

struct sit
{
    int s,w,tim,dir;
    sit(int a,int b,int d,int c)
    {
        s = a,w = b,tim = c,dir = d;
    }
};

int vis[202][202][2];

int bfs(int x,int y,int n)
{
    queue<sit> q;
    q.push(sit(x,y,0,0));
    while(!q.empty())
    {
        if(q.front().dir == 1 && q.front().s == x && q.front().w == y)
            return q.front().tim;
        int shep = q.front().s,wof = q.front().w;
        for(int i = 0;i <= shep;i++)
        {
            for(int j = 0;j <= wof;j++)
            {
                if(i == j && j == 0)
                    continue;
                if(i + j > n || (j > i && i != 0) || (wof - j > shep - i && shep != i))
                    continue;
                if(x - shep + i < y - wof + j && x - shep + i != 0)
                    continue;
                if(vis[x - shep + i][y - wof + j][q.front().dir ^ 1])
                    continue;
                vis[x - shep + i][y - wof + j][q.front().dir ^ 1] = 1;
                q.push(sit(x - shep + i,y - wof + j,q.front().dir ^ 1,q.front().tim + 1));
            }
        }
        q.pop();
    }
    return -1;
}

int main()
{
    int x,y,n;
    while(cin >> x >> y >> n)
    {
        memset(vis,0,sizeof(vis));
        vis[x][y][0] = 1;
        cout << bfs(x,y,n) << endl;
    }
    return 0;
}
