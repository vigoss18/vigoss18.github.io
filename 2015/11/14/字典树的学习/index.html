字典树的学习

发表于 2015-11-14 | 分类于 ACM |

昨天的模拟赛中做到了一道字典树的题，钻研了几个小时感觉蛮有意思的，今天就练几道字典树的题吧，学习一下，学习一下。

hdu5536

题意是给你n个数(1 <= n <= 1000)，找出三个不同的数，满足(ai + aj) ^ ak最大，暴力枚举的话复杂度是n^3，会超时，那么就想办法优化到n^2，然而并不会，那就n^3剪枝，然而剪枝超时，gg，好叭，那就只能学新知识（字典树）了0.0。

字典树嘛，就是查询起来非常方便的一种树，据说效率比哈希表还高啊。建树过程也很简单，就不赘述了，开始练手。

#include<bits/stdc++.h>
#define ll long long
#define MAXN 200050
#define INF 9999999
using namespace std;

int sz,wtf[2];                          //wtf数组用来标记离根最远的节点位置

struct Trie
{
    int next[2];
    int flag,cnt;
    void init()
    {
        next[0] = -1,next[1] = -1;
        flag = -1;                          //flag用来标记第一次出现这个数的位置
        cnt = 0;                            //cnt用来标记这个数出现了几次
    }
};

Trie trie[1024 << 6];
int n,a[1024],maxx = 0;

void init()
{
    trie[0].init();
    maxx = 0;
    sz = 1;
}

void InsertTrie(int num,int pos)
{
    int p = 0;
    for(int i = 30;i >= 0;i--)
    {
        int ch = (num >> i) & 1;
        if(trie[p].next[ch] == -1)
        {
            trie[p].next[ch] = sz;
            trie[sz++].init();
        }
        p = trie[p].next[ch];
    }
    if(trie[p].flag == -1)
    {
        trie[p].flag = pos;
    }
    trie[p].cnt++;
}


int query(int n,int t1,int t2,int tim)
{
    int j = -1,ret = 0,p = 0;
    wtf[tim] = -1;
    for(int i = 30;i >= 0;i--)
    {
        int ch = (n >> i) & 1;
        for(j = tim - 1;j >= 0;j--)
        {
            if(wtf[j] == p)
            {
                break;
            }
        }
        if(j != -1)
        {
            p = trie[p].next[ch];
            continue;
        }
        if(trie[p].next[0] != -1 && trie[p].next[1] != -1)
        {
            wtf[tim] = p;
        }
        if(trie[p].next[!ch] != -1)
        {
            p = trie[p].next[!ch];
            ret |= (1 << i);
        }
        else
        {
            p = trie[p].next[ch];
        }
    }
    int us = a[trie[p].flag];               //判断此ret的合法性
    if(a[t1] == a[t2] && a[t1] == us && trie[p].cnt > 2)
        return ret;
    if(((a[t1] == us) ^ (a[t2] == us)) && trie[p].cnt > 1)
        return ret;
    if(a[t1] != us && a[t2] != us)
        return ret; 
    return -1;
}

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        init();
        cin >> n;
        for(int i = 0;i < n;i++)
        {
            cin >> a[i];
            InsertTrie(a[i],i);
        }
        for(int i = 0;i < n;i++)
        {
            for(int j = i + 1;j < n;j++)
            {
                int tmp = query(a[i] + a[j],i,j,0);
                if(tmp == -1)
                    tmp = query(a[i] + a[j],i,j,1);
                if(tmp == -1)
                    tmp = query(a[i] + a[j],i,j,2);
                maxx = max(maxx,tmp);
            }
        }
        cout << maxx << endl;
    }
    return 0;
}

先把输入的全部数字转化成二进制后存为字典树，然后n^2枚举i和j，每次查询到使(ai + aj) ^ qes最大的qes，当然要判合法性，不能重复使用同一个数字，考虑到可能输入同一个数字多次，那么就在建树的时候标记一下这个数出现的次数，同时记录下第一次出现这个数的位置。然后开始查询的时候就有一种类似深搜的思想，每次优先找这一位的相反值是否存在，如果不存在，那么就回去，回到的节点当然是离根最远的一个点（且有岔路），就用wtf数组来标记上一次搜索后离根最远且存在岔路的节点。蛤，这样就可以过啦，来人啊，下一题！！！

hdu1247

#include<bits/stdc++.h>
#define ll long long
#define MAXN 2333333
#define INF 9999999
using namespace std;

struct Trie
{
    int next[26];
    int word[26];
    void init()
    {
        memset(next,-1,sizeof(next));
        memset(word,-1,sizeof(word));
    }
};

Trie trie[MAXN];
int sz;

void insertTrie(string s)
{
    int p = 0;
    for(int i = 0;i < s.size();i++)
    {
        if(trie[p].next[s[i] - 'a'] == -1)
        {
            trie[p].next[s[i] - 'a'] = sz;              //因为这两句wa了好多次！！第二句的时候才能sz++
            trie[sz++].init();
        }
        p = trie[p].next[s[i] - 'a'];
    }
    trie[p].word[s[s.size() - 1] - 'a'] = 1;
}

bool query(string s1,string s2)
{
    int p = 0;
    for(int i = 0;i < s1.size();i++)
    {
        if(trie[p].next[s1[i] - 'a'] == -1)
            return false;
        p = trie[p].next[s1[i] - 'a'];
    }
    if(trie[p].word[s1[s1.size() - 1] - 'a'] != 1)
        return false;
    p = 0;
    for(int i = 0;i < s2.size();i++)
    {
        if(trie[p].next[s2[i] - 'a'] == -1)
            return false;
        p = trie[p].next[s2[i] - 'a'];
    }
    if(trie[p].word[s2[s2.size() - 1] - 'a'] == -1)
        return false;
    return true;
}

void init()
{
    sz = 1;
    memset(trie[0].next,-1,sizeof(trie[0].next));
    memset(trie[0].word,-1,sizeof(trie[0].word));
}

int main()
{
    string s[50005];
    int n = 0;
    init();
    while(cin >> s[n++])
    {
        insertTrie(s[n - 1]);
    }
    n--;
    string p[50005];
    int cnt = 0;
    for(int i = 0;i < n;i++)
    {
        for(int j = 1;j <= s[i].size() - 1;j++)
        {
            string s1 = s[i].substr(0,j);
            string s2 = s[i].substr(j,s[i].size() - j);
            if(query(s1,s2) || query(s2,s1))
            {
                p[cnt++] = s[i];
                break;
            }
        }
    }
    sort(p,p + cnt);
    for(int i = 0;i < cnt;i++)
    {
        cout << p[i] << endl;
    }
    //cout << endl;
    return 0;
}

蛤，题意是给你n串字符串，让你输出能由输入的字符串中的两串并起来合成的字符串，先把所有的字符串存入到字典树中，用word数组表示在这个位置上结束了一个单词，然后依次遍历每一个字符，枚举所有可能的状态并进行查询即可，wa了好久的原因竟然是因为自己在建树的时候的地址加早了ORZZ不能再翻这个错了，来人啊！下一题！！

#include<bits/stdc++.h>
#define ll long long
#define MAXN 6333333
#define INF 9999999
using namespace std;

struct Trie
{
    int next[11];
    int num,flag;
    void init()
    {
        memset(next,-1,sizeof(next));
        num = -1;
        flag = -1;
    }
};

Trie trie[MAXN];
int sz;
int f1[65],f2[65],f3[65];

void InsertTrie(string s,int nn)
{
    int p = 0;
    for(int i = 0;i < s.size();i++)
    {
        if(trie[p].next[s[i] - '0'] == -1)
        {
            trie[p].next[s[i] - '0'] = sz;
            trie[sz++].init();
        }
        p = trie[p].next[s[i] - '0'];
        if(trie[p].flag == -1)
            trie[p].num = nn,trie[p].flag = 1;
    }

}

int query(string s)
{
    int p = 0;
    for(int i = 0;i < s.size();i++)
    {
        if(trie[p].next[s[i] - '0'] == -1)
            return -1;
        p = trie[p].next[s[i] - '0'];
    }    
    return trie[p].num;
}

void init()
{
    sz = 1;
    trie[0].init();
    memset(f1,0,sizeof(f1));
    memset(f2,0,sizeof(f2));
    memset(f3,0,sizeof(f3));
    f1[0] = 1,f2[0] = 1;
    InsertTrie("1",0);
    for(int i = 2;i < 100000;i++)
    {
        int pl = 0;
        for(int j = 0;j <= 60;j++)
        {
            f3[j] = f1[j] + f2[j] + pl;
            pl = f3[j] / 10;
            f3[j] %= 10;
        }
        int h;
        for(int j = 59;j >= 0;j--)
        {
            if(f3[j])
            {
                h = j;
                break;
            }
        }
        string s;
        for(int j = h,k = 0;j >= 0 && k < 40;j--,k++)
        {
            s += (char)(f3[j] + '0');
        }
        InsertTrie(s,i);
        if(h > 55)
        {
            for(int j = 1;j < 59;j++)
            {
                f3[j - 1] = f3[j];
                f2[j - 1] = f2[j];
            }
        }
        for(int j = 0;j < 60;j++)
        {
            f1[j] = f2[j];
        }
        for(int j = 0;j < 59;j++)
        {
            f2[j] = f3[j];
        }
    }
}

int main()
{
    init();
    int t;
    cin >> t;
    for(int i = 1;i <= t;i++)
    {
        string s;
        cin >> s;
        printf("Case #%d: %d\n",i,query(s));
    }
    return 0;
}
