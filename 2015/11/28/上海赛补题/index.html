##Kingdom of Black and White
水题，判断修改一个数字后的分割长度的平方和的最大值,暴力大法好

#include <bits/stdc++.h>
#define ll long long
#define eps 1e-6
#define PI acos(-1.0)
using namespace std;

int main()
{
    int t;
    cin >> t;
    int cas = 1;
    while(t--)
    {
        string s;
        cin >> s;
        ll p[100005];
        int cnt = 0;
        memset(p,0,sizeof(p));
        p[0] = 1;
        for(int i = 1;i < s.size();i++)
        {
            if(s[i] != s[i - 1])
                p[++cnt] = 1;
            else
                p[cnt]++;
        }
        ll sum = 0,maxx = 0;
        printf("Case #%d: ",cas++);
        for(int i = 0;i <= cnt;i++)
        {
            sum += p[i] * p[i];
        }
        maxx = sum;
        if(cnt == 0)
            cout << sum << endl;
        else
        {
            for(int i = 0;i <= cnt;i++)
            {
                ll us = 0;
                if(i == 0)
                {
                    us = sum - 2 * p[0] + 1 + 2 * p[i + 1] + 1;
                }
                else if(i == cnt)
                {
                    us = sum - 2 * p[i] + 1 + 2 * p[i - 1] + 1;
                }
                else
                {
                    if(p[i] == 1)
                    {
                        us = sum - 1 - p[i - 1] * p[i - 1] - p[i + 1] * p[i + 1] + (p[i - 1] + p[i + 1] + 1) * (p[i - 1] + p[i + 1] + 1);
                    }
                    else
                    {
                        if(p[i - 1] > p[i - 1])
                        {
                            us = sum - 2 * p[i] + 1 + p[i - 1] * 2 + 1;
                        }
                        else
                        {
                            us = sum - 2 * p[i] + 1 + p[i + 1] * 2 + 1;
                        }
                    }
                }
                maxx = max(maxx,us);
            }cout << maxx << endl;
        }

    }
    return 0;
}

##LCM Walk
当时比赛的时候过的第三道题，题意是从一个起点(x,y)出发，每次运动后会到(x + LCM(x,y),y)或者(x,y + LCM(x + y)),现在给你终点，问你有多少个起点。比赛的时候我想着dfs（设x小y大），每次一定是大的数来减少一个lcm，那么我就枚举，枚举减少的lcm值，，如果满足lcm(x,y - t) = t,那就+1然后往里深搜，当时……确实没有超时来着，但是在hdoj做的时候就T了，然后百度了一下题解，发现原来是一道数学题。。。
之前的一个点应该为(x,y’ + z);z = LCM(x,y’);z = k x;如果t == gcd(x,y),那么t == gcd(x,y’);
两数之间的gcd和lcm相乘即为两数相乘，那么x (y - z) == k t;推出z = (x y) / (x + t);接下来就好写多了。

#include <bits/stdc++.h>
#define ll long long
#define eps 1e-6
#define MOD 10007
using namespace std;

ll gcd(ll a,ll b)
{
    return (a == 0 ? b : gcd(b % a,a));
}

ll lcm(ll a,ll b)
{
    return a / gcd(a,b) * b;
}

int main()
{
     int t,cas = 1;
     cin >> t;
     while(t--)
     {
        ll a,b,ans = 1;
        cin >> a >> b;
        if(a > b)
            swap(a,b);
        ll z = (a * b) / (a + gcd(a,b));
        while(z >= a && z < b && z != b && z % (b - z) == 0 && z % a == 0)
        {
            b -= z;
            if(a > b)
                swap(a,b);
            z = (a * b) / (a + gcd(a,b));
            ans++;
            //cout << a  <<  " " << b << endl;
        }
        printf("Case #%d: ",cas++);
        cout << ans << endl;
     }
    return 0;
}

##Binary Tree
给你一颗完全二叉树，问你遍历到第n层时如何操作才能找到m，操作方法是左右节点二选一且选择加减。
当时死活没想到做法啊，q巨说只要一直走最左边，然后判断哪些用+哪些用-就可以了，最后一层可以选择是走左边还是右边。

现在思考一下，其实这就是一道sb题啊……一直走最左边的话，取到的数就是1，2，4，8……接下来只要用二进制的思想来构造你所要求的那个数就可以了啊。我的想法是先把前面的数全部取-，最后一个数取正，那么此时构造的数一定是1或者2，用来控制最后两位，其余的位数，我们可以这么想，比如一个数1010，8的位置上是1那么就是由4变化而来（-4变成+4），所以只要先把要求的数转化成二进制，然后通过判断当前位是不是1来确定到底是取负还是取正就可以了
