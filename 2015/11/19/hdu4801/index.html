hdu4801

题意很简单，给你一个二阶魔方的初始状态，让你判断在转的次数少于n的情况下，最多能有几个面是相同的。因为自己平时喜欢玩魔方，所以我觉得自己很快地就推出了魔方的状态改变只有四种情况，然而我这个sb忘记了其实是有六种情况的，而且一面的顺逆时针就已经等于另一面的旋转了，那么我就只要退出三个状态就够了，然后我就在这三个状态的表中各种找错，各种错啊，那个惨啊，我是用pair来存储置换状态，那么只要把first和second换一下就是换了一个方向来转了，pair写的也不够简洁，等下去修改。

#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
#include<queue>
#define ll long long
using namespace std;

pair<int,int>turn[3][22];
int n,ans;

struct cube
{
    int f[24],tim;
};

int al[6][4];

void init()             //pair学习不精啊，应该用make_pair来写更加简单，下次注意
{
    turn[0][0].first = 0,turn[0][0].second = 20,turn[0][1].first = 2,turn[0][1].second = 22;
    turn[0][2].first = 6,turn[0][2].second = 0,turn[0][3].first = 12,turn[0][3].second = 2;
    turn[0][4].first = 16,turn[0][4].second = 6,turn[0][5].first = 18,turn[0][5].second = 12;
    turn[0][6].first = 20,turn[0][6].second = 16,turn[0][7].first = 22,turn[0][7].second = 18;
    turn[0][8].first = 11,turn[0][8].second = 5,turn[0][9].first = 5,turn[0][9].second = 4;
    turn[0][10].first = 4,turn[0][10].second = 10,turn[0][11].first = 10,turn[0][11].second = 11;

    turn[1][0].first = 2,turn[1][0].second = 8,turn[1][1].first = 3,turn[1][1].second = 14;
    turn[1][2].first = 8,turn[1][2].second = 17,turn[1][3].first = 14,turn[1][3].second = 16;
    turn[1][4].first = 17,turn[1][4].second = 11,turn[1][5].first = 16,turn[1][5].second = 5;
    turn[1][6].first = 11,turn[1][6].second = 2,turn[1][7].first = 5,turn[1][7].second = 3;
    turn[1][8].first = 6,turn[1][8].second = 7,turn[1][9].first = 7,turn[1][9].second = 13;
    turn[1][10].first = 13,turn[1][10].second = 12,turn[1][11].first = 12,turn[1][11].second = 6;//check

    turn[2][0].first = 9,turn[2][0].second = 7,turn[2][1].first = 8,turn[2][1].second = 6;
    turn[2][2].first = 7,turn[2][2].second = 5,turn[2][3].first = 6,turn[2][3].second = 4;
    turn[2][4].first = 5,turn[2][4].second = 22,turn[2][5].first = 4,turn[2][5].second = 23;
    turn[2][6].first = 22,turn[2][6].second = 9,turn[2][7].first = 23,turn[2][7].second = 8;
    turn[2][8].first = 3,turn[2][8].second = 2,turn[2][9].first = 1,turn[2][9].second = 3;
    turn[2][10].first = 0,turn[2][10].second = 1,turn[2][11].first = 2,turn[2][11].second = 0;
}

int check(cube c)
{
    int re = 0;
    if(c.f[0] == c.f[1] && c.f[1] == c.f[2] && c.f[2] == c.f[3] && c.f[3] == c.f[0])
        re++;
    if(c.f[4] == c.f[5] && c.f[5] == c.f[10] && c.f[10] == c.f[11] && c.f[11] == c.f[4])
        re++;
    if(c.f[6] == c.f[7] && c.f[7] == c.f[12] && c.f[12] == c.f[13] && c.f[13] == c.f[6])
        re++;
    if(c.f[8] == c.f[9] && c.f[9] == c.f[14] && c.f[14] == c.f[15] && c.f[15] == c.f[8])
        re++;
    if(c.f[16] == c.f[17] && c.f[17] == c.f[18] && c.f[18] == c.f[19] && c.f[19] == c.f[16])
        re++;
    if(c.f[20] == c.f[21] && c.f[21] == c.f[22] && c.f[22] == c.f[23] && c.f[23] == c.f[20])
        re++;
    return re;
}

void bfs(cube c)
{
    queue<cube>q;
    q.push(c);
    while(!q.empty())
    {
        ans = max(ans,check(q.front()));
        if(q.front().tim >= n)
        {
            q.pop();
            continue;
        }
        if(ans == 6)
            break;
        cube tmp = q.front();
        q.pop();
        for(int i = 0;i < 3;i++)
        {
            cube nxt;
            for(int j = 0;j < 24;j++)
            {
                nxt.f[j] = tmp.f[j];
            }
            nxt.tim = tmp.tim + 1;
            for(int j = 0;j < 12;j++)
            {
                nxt.f[turn[i][j].first] = tmp.f[turn[i][j].second];
            }
            q.push(nxt);
            cube nxt1;
            for(int j = 0;j < 24;j++)
            {
                nxt1.f[j] = tmp.f[j];
            }
            nxt1.tim = tmp.tim + 1;
            for(int j = 0;j < 12;j++)
            {
                nxt1.f[turn[i][j].second] = tmp.f[turn[i][j].first];
            }
            q.push(nxt1);
        }
    }
}

int main()
{
    init();
    //freopen("t1.txt","w",stdout);
    //freopen("1.txt","r",stdin);
    while(cin >> n)
    {
        cube c;
        for(int i = 0;i < 24;i++)
            cin >> c.f[i];
        c.tim = 0;
        ans = check(c);
        bfs(c);
        cout<< ans << endl;
    }
    return 0;
}
