##A and B Problem
给你一个AB串，看成环，问最少经过几次交换之后就可以使得AB串分开。
当时异想天开的想直接暴力一发，先算出AB的数量，用少的数量作为截取串的长度，然后线性遍历取最优解即可。

#include <bits/stdc++.h>
#define mod 1e9+7
#define ll long long
using namespace std;

int main(){
    int t,cnt = 1;
    cin >> t;
    while(t--)
    {
        string s;
        cin >> s;
        printf("Case %d: ",cnt++);
        int f[2] = {0};
        for(int i = 0;i < s.length();i++)
        {
            f[s[i] - 'A']++;
        }
        int sum = 0,maxx = 0;
        if(f[0] < f[1])
        {
            for(int i = 0;i < f[0];i++)
            {
                if(s[i] == 'A')
                    sum++;
            }
            s += s.substr(0,f[0]);
            for(int ed = f[0],st = 0;ed < s.length();ed++,st++)
            {
                if(s[st] == 'A')
                    sum--;
                if(s[ed] == 'A')
                    sum++;
                maxx = max(maxx,sum);
            }
            printf("%d\n",f[0] - maxx);
        }
        else
        {
            for(int i = 0;i < f[1];i++)
            {
                if(s[i] == 'B')
                    sum++;
            }
            s += s.substr(0,f[1]);
            for(int ed = f[1],st = 0;ed < s.length();ed++,st++)
            {
                if(s[st] == 'B')
                    sum--;
                if(s[ed] == 'B')
                    sum++;
                maxx = max(maxx,sum);
            }
            printf("%d\n",f[1] - maxx);
        }
    }
    return 0;
}

##Squarefree number
给你一个数，让你求这个数能不能被一个完全平方数整除。这样的数都是可以化为A^2*B的形式，那么里面的每一个因子的大小都不能超过1e6，然后暴力就可以了。比赛的时候死活不能ac，结果最后发现是模板里那句关闭同步流的语句没有删掉，郁闷啊！！！！

#include <cstdio>
#include <algorithm>
#include <iostream>

#include <cstring>

const int MAXV = 2 * 1e6; 
bool isPrime[MAXV+1]; 
int prime[MAXV+1]; 
int size=0; 
void getprime()  
{  
    memset(isPrime, true, sizeof(isPrime));
    int sq = sqrt((double)MAXV) + 1; 
    int i,j,k;  
    for(i = 2;i <= sq; i++)  
        if(isPrime[i])  
            for(j = 2,k = MAXV/i+1;j < k;j++)  
                isPrime[i*j] = false;  
    for( i = 2 ; i <= MAXV; i++)  
        if(isPrime[i])    
            prime[size++] = i;
    isPrime[0] = isPrime[1] = false;
}  
int main()
{
    getprime();
    int t,cas = 1;
    cin >> t;
    while(t--)
    {
        long long s,ss;
        cin >> s;
        ss = s;
        int ok =0;
        for(int i = 0;i < size;i++){
            int cnt = 0;
            while(s % prime[i] == 0)
            {
                cnt++;
                if(cnt == 2){
                    ok = 1;
                    break;
                }
                s /= prime[i];
            }
            if(s == 1)
                break;
        }
        if(!ok)
        {
        for(int i = 1;i < 1000000;i++)
        {
            if(i >= ss)
                break;
            if(ss % i == 0)
            {
                long long sss = ss / i;
                double cn = sqrt((double)sss);
                if(cn - (int)(cn) < 0.000001)
                {
                    ok = 1;
                    break;
                }
            }
        }}
        printf("Case %d: ",cas++);
        if(ok)
            cout << "No\n";

return 0;
