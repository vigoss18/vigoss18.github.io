今天开始刷51NOD上的基本题，感觉题目很科学啊。今天做的是一道SPFA算法的题目，可以用来求单源最短路。感觉不算难吧。

学习网址

模板上的代码非常易懂啊，不过我自己写的时候没有用这样的写法，是用vector来建图的，感觉这样容易理解一些。

题目地址

#include <bits/stdc++.h>
#define SIZE 1005
#define ll long long
using namespace std;

int n,m,st,ed,val[SIZE];

struct node
{
    int v,tim;
    node(int vv,int timee)
    {
        v = vv,tim = timee;
    }
};

vector<node>g[SIZE];

void SPFA()
{
    queue<int>q;
    int ans[SIZE],vis[SIZE],d[SIZE];
    memset(ans,0,sizeof(ans));
    memset(vis,0,sizeof(vis));
    memset(d,1,sizeof(d));
    d[st] = 0,vis[st] = 1,q.push(st);
    while(!q.empty())
    {
        int u = q.front();
        vis[u] = 0;
        q.pop();
        for(int i = 0;i < g[u].size();i++)
        {
            int te = g[u][i].v;
            int tm = g[u][i].tim;
            if(d[te] > tm + d[u])
            {
                d[te] = tm + d[u];
                ans[te] = ans[u] + val[te];
                if(!vis[te])
                {
                    vis[te] = 1;
                    q.push(te);
                }
            }
            else if(d[te] == tm + d[u])
            {
                if(ans[te] < ans[u] + val[te])
                {
                    ans[te] = ans[u] + val[te];
                    if(!vis[te])
                    {
                        vis[te] = 1;
                        q.push(te);
                    }
                }
            }
        }
    }
    printf("%d %d\n",d[ed],ans[ed] + val[st]);
}

int main()
{
    while(cin >> n >> m >> st >> ed)
    {
        for(int i = 0;i < n + 1;i++)
        {
            g[i].clear();
        }
        for(int i = 0;i < n;i++)
        {
            cin >> val[i];
        }
        for(int i = 0;i < m;i++)
        {
            int u,v,t;
            scanf("%d %d %d",&u,&v,&t);
            g[u].push_back(node(v,t));
            g[v].push_back(node(u,t));
        }
        SPFA();
    }
    return 0;
}

也是一道裸的单源最短路，题意是有n个点，有n-1个人要从点1出发，到第i个点，然后回来，求所有距离的最短路，先建立一个正向的图，做一次SPFA，然后把所有的路反过来再做一次就行了，水过。

#include<bits/stdc++.h>
#define ll long long
#define MAXN 1000001
#define eps 1e-6
using namespace std;

struct node
{
    int v;
    int len;
    node(int a,int b)
    {
        v = a,len = b;
    }
};

int n,m;
vector<node>g[MAXN];
int vis[MAXN],d[MAXN],a[MAXN],b[MAXN],c[MAXN];

void SPFA()
{
    for(int i = 0;i <= n;i++)
        d[i] = 1000000000;
    memset(vis,0,sizeof(vis));
    queue<int>q;
    q.push(1);
    vis[1] = 1,d[1] = 0;
    while(!q.empty())
    {
        int nod = q.front();
        q.pop();
        vis[nod] = 0;
        for(int i = 0;i < g[nod].size();i++)
        {
            int nxtnod = g[nod][i].v;
            if(d[nxtnod] > d[nod] + g[nod][i].len)
            {
                d[nxtnod] = d[nod] + g[nod][i].len;
                if(!vis[nxtnod])
                {
                    vis[nxtnod] = 1;
                    q.push(nxtnod);
                }
            }
        }
    }
}

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        cin >> n >> m;
        for(int i = 0;i <= n;i++)
        {
            g[i].clear();
        }
        for(int i = 0;i < m;i++)
        {
            scanf("%d %d %d",&a[i],&b[i],&c[i]);
            g[a[i]].push_back(node(b[i],c[i]));

        }
        SPFA();
        int sum1 = 0;
        for(int i = 2;i <= n;i++)
            sum1 += d[i];
        for(int i = 1;i <= n;i++)
            g[i].clear();
        for(int i = 0;i < m;i++)
            g[b[i]].push_back(node(a[i],c[i]));
        SPFA();
        int sum2 = 0;
        for(int i = 2;i <= n;i++)
            sum2 += d[i];
        cout << sum1 + sum2 << endl;
    }
    return 0;
}

噫，也算是不难的求最短路吧，题意是给你一个滑雪场，有两种边，一种是向上走，一种是向下滑，问你要找到两个点，从a点出发到b点的时间为t1,从b点出发到a点的时间是t2，要求t1/t2最大，只要做2*n次最短（长）路就可以了哇，然后枚举一下，打印路径倒是让我脑子混乱了一会儿。过了样例之后一发wa，题中写道1.3555应该向上取整到1.356，所以我+了一个0.0005，其实不用的，去掉之后就ac了。

int pre[MAXN][MAXN],pre1[MAXN][MAXN];

void SPFA(int x)
{

    q.push(x);

void path(int st,int ed)
{
    int ff[MAXN],cnt1 = 0,fff[MAXN],cnt2 = 0;
    ff[cnt1++] = ed;
    int us = ed;
    while(us != st)
    {
        ff[cnt1++] = pre1[st][us];
        us = pre1[st][us];
    }
    us = st;
    fff[cnt2++] = st;
    while(us != ed)
    {
        fff[cnt2++] = pre[ed][us];
        us = pre[ed][us];
    }
    for(int i = cnt1 - 1;i > 1;i--)
        cout << ff[i] << " ";
    cout << ff[1];
    for(int i = cnt2 - 1;i >= 0;i--)
        cout << " " << fff[i];
    cout << endl;
}
