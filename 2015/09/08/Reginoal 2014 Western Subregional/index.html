##A·Arithmetic Derivative
p′=1对于任意质数

(ab)′=a′b+ab′对于任意自然数a,b

给你一个数，让你求s′,分解质因数就行，要先打个质数表啊！现场做的时候打了一个莫名其妙的质数表，然后就心累了，然后就再也没写出来了……

#include<bits/stdc++.h>
#define LL long long
#define MOD 1000000007
#define PI acos(-1.0)
using namespace std;

const int MAXV = 1e5; 
bool isPrime[MAXV+1]; 
int prime[MAXV+1]; 
int size=0; 
void getprime()  
{  
    memset(isPrime, true, sizeof(isPrime));
    int sq = sqrt((double)MAXV) + 1; 
    int i,j,k;  
    for(i = 2;i <= sq; i++)  
        if(isPrime[i])  
            for(j = 2,k = MAXV/i+1;j < k;j++)  
                isPrime[i*j] = false;  
    for( i = 2 ; i <= MAXV; i++)  
        if(isPrime[i])    
            prime[size++] = i;
    isPrime[0] = isPrime[1] = false;
}  

int main()
{
    getprime();
    int N;
    cin >> N;
    while(N--)
    {
        int sv[10001],cnt = 0;
        int u;
        int sum = 1;
        cin >> u;
        for(int i = 0;i < size;i++)
        {
            while(u % prime[i] == 0)
            {
                sv[cnt++] = prime[i];
                u /= prime[i];
                sum *= prime[i];
            }
            if(u == 1)
                break;
        }
        long long ou = 0;
        for(int i =0;i < cnt;i++)
        {
            ou += sum / sv[i];
        }
        cout << ou << endl;
    }
return 0;
}
