昨天模拟赛做到的一道题目，是三月份的月赛的题，感觉可以啊……当时想歪了……

题目链接：Buy Cakes

题意是有n块蛋糕，k张折扣券，m块钱，每个蛋糕有对应的原价和折扣价，问你最多能买几块蛋糕。

一开始我想的时候是觉得先把按原价排个序尽可能多的买完之后，把剩下的蛋糕按照折扣价排序，已经准备买的答案按照差价排序，尽可能地把已经准备买的蛋糕里能换的差价最大的换了，看看能不能再多买点蛋糕。经过和宝树讨论之后我又觉得，应该把已经准备买的蛋糕按原价排序，扔出一个原价蛋糕后尽可能多地用折扣券买蛋糕。
但是我还是没有理解贪心的本质啊，在用光了折扣券的前提下贪心才是正确的。那么我们先按折扣价排序后尽可能多的买，把他们存到一个按照折扣价排序的优先队列，剩余的蛋糕先按照原价存到一个另一个优先队列里，同时再存到一个按照差价排序的优先队列里，对于当前状态，要么扔掉原价最高的蛋糕，要么扔掉折扣价最高的蛋糕且在原价买的蛋糕里把差价最大的蛋糕改用折扣券买，比较一下哪个减少的价钱多就取哪个，扔到总价格小于m为止。

#include <bits/stdc++.h> 
using namespace std;
typedef long long ll;
#define PB push_back
#define UB upper_bound
#define LB lower_bound
#define MP make_pair
#define pii pair<int,int>
#define vi vector<int>
#define vii vector<vi>
#define mst(x,y) memset(x,y,sizeof(x))
#define fr(x) freopen(x,"r",stdin)
#define fw(x) freopen(x,"w",stdout)
#define iin(x) scanf("%d",&x)
#define sp system("pause")
#define INF 1e9
#define MAXN 100005
#define PI acos(-1.0)
#define eps 1e-10
#define pll pair<ll,ll>
const ll MOD = 1e9 + 7;

struct both
{
    int num;ll a,b;
    both(){}
    both(int q,ll w,ll e)
    {
        num = q,a = w,b = e;
    }
}node[MAXN];

struct yuan
{
    int num;ll a;
    yuan(int q,ll w)
    {
        num = q,a = w;
    }
    friend bool operator < (yuan a,yuan b)
    {
        return a.a < b.a;
    }
};

struct zhe
{
    ll b;
    zhe(ll w)
    {
        b = w;
    }
    friend bool operator < (zhe a,zhe b)
    {
        return a.b < b.b;
    }
};

bool cmp1(both a,both b)
{
    return a.b < b.b;
}

bool cmp2(both a,both b)
{
    return a.a < b.a;
}

bool cmpp(both a,both b)
{
    return (a.a - a.b) > (b.a - b.b);
}

struct cmp
{
    bool operator()(both a,both b)
    {
        return (a.a - a.b) < (b.a - b.b);
    }
};

ll n,m,k;

void work()
{
    int ok = 0;
    if(k == 0)
        sort(node,node + n,cmp2);
    else
        sort(node,node + n,cmp1);
    for(int i = 0;i < n;i++)
    {
        if(k == 0)
        {
            if(m >= node[i].a)
                m -= node[i].a;
            else
            {
                cout << i << endl;ok = 1;
                break;
            }
        }
        else
        {
            if(m >= node[i].b)
                m -= node[i].b;
            else
            {
                cout << i << endl;ok = 1;
                break;
            }
        }
    }
    if(!ok)
        cout << n << endl;
}

priority_queue<zhe>q1;
priority_queue<yuan>q2;
priority_queue<both,vector<both>,cmp>q3;

void init()
{
    while(!q1.empty())
        q1.pop();
    while(!q2.empty())
        q2.pop();
    while(!q3.empty())
        q3.pop();
}

int main()   
{   
    int t;
    cin >> t;
    while(t--)
    {
        init();
        cin >> n >> k >> m;
        for(int i = 0;i < n;i++)
        {
            ll a,b;
            scanf("%lld %lld",&a,&b);
            node[i] = both(i,a,b);
        }
        if(k == 0 || k == n)
            work();
        else
        {
            ll ans = 0;
            sort(node,node + n,cmpp);
            for(int i = 0;i < k;i++)
            {
                ans += node[i].b;
                q1.push(zhe(node[i].b));
            }
            for(int i = k;i < n;i++)
            {
                ans += node[i].a;
                q2.push(yuan(i,node[i].a));

                q3.push(both(i,node[i].a,node[i].b));
            }
            int vis[MAXN];mst(vis,0);
            while(ans > m)
            {
                while(!q2.empty() && vis[q2.top().num])
                    q2.pop();
                while(!q3.empty() && vis[q3.top().num])
                    q3.pop();
                if(q2.empty())//所有的原价都扔光了
                {
                    n--;ans -= q1.top().b;
                    q1.pop();
                }
                if(!q2.empty() && !q3.empty())
                {
                    ll x = q2.top().a,y = q1.top().b + q3.top().a - q3.top().b;
                    if(x > y)
                    {
                        ans -= x;
                        vis[q2.top().num] = 1;
                        q2.pop();
                    }
                    else
                    {
                        ans -= y;
                        vis[q3.top().num] = 1;
                        q1.pop();
                        q1.push(zhe(q3.top().b));
                        q3.pop();
                    }
                    n--;
                }    
            }
            cout << n << endl;
        }
    }
}
