给你一个只包含#和的矩形，求有多少个矩形包含了至少m个。
数据范围很小……暴力

#include <bits/stdc++.h> 
using namespace std;
typedef long long ll;
#define PB push_back
#define UB upper_bound
#define LB lower_bound
#define MP make_pair
#define pii pair<int,int>
#define vi vector<int>
#define vii vector<vi>
#define mst(x,y) memset(x,y,sizeof(x))
#define fr(x) freopen(x,"r",stdin)
#define fw(x) freopen(x,"w",stdout)
#define iin(x) scanf("%d",&x)
#define sp system("pause")
#define INF 1e9
#define MAXN 1000005
#define PI acos(-1.0)
#define eps 1e-10
const ll MOD = 1e9 + 7;

int main()   
{   
    int r,c,n,p,mp[11][11];
    mst(mp,0);
    int hh = 0;
    cin >> r >> c >> n >> p;
    for(int i = 0;i < n;i++)
    {
        int a,b;
        cin >> a >> b;
        mp[a - 1][b - 1] = 1;
    }
    for(int i = 0;i < r;i++)
    {
        for(int j = 0;j < c;j++)
        {
            for(int a = i;a < r;a++)
            {
                for(int b = j;b < c;b++)
                {
                    int sum = 0;
                    for(int q = i;q <= a;q++)
                    {
                        for(int k = j;k <= b;k++)
                        {
                            sum += mp[q][k];
                        }
                    }
                    if(sum >= p)
                        hh++;
                }
            }
        }
    }
    cout << hh << endl;
}

C. XOR Equation

给出n和m求有多少对数(a,b)使得a+b=n，且a^b=m。
分析一下，a^b要等于m那么最基本的可以使a=m,b=0,同时为了满足a+b=n，就只能在a为1的二进制位都更改为1，XD于是题目一下就变得简单了。

#include <bits/stdc++.h> 
using namespace std;
typedef long long ll;
#define PB push_back
#define UB upper_bound
#define LB lower_bound
#define MP make_pair
#define pii pair<int,int>
#define vi vector<int>
#define vii vector<vi>
#define mst(x,y) memset(x,y,sizeof(x))
#define fr(x) freopen(x,"r",stdin)
#define fw(x) freopen(x,"w",stdout)
#define iin(x) scanf("%d",&x)
#define sp system("pause")
#define INF 1e9
#define MAXN 200005
#define PI acos(-1.0)
#define eps 1e-10
const ll MOD = 1e9 + 7;

int main()   
{   
    ll a,b,ok = 0;
    cin >> a >> b;
    if(a == b)
        ok = 1;
    ll c = (a - b) / 2;
    if(c < 0 || (2 * c + b != a) || (c & b))
    {
        cout << 0 << endl;
        return 0;
    }
    ll bits = 0;
    while(b)
    {
        if(b & 1)
            bits++;
        b >>= 1;
    }
    ll ou = 1LL << bits;
    if(ok)
        ou -= 2;
    cout << ou << endl;
}

D. Factory Repairs

有一台机器，修理之前每天能生产a个物品，修理之后每天能生产b个物品，修理持续k天，输入p个操作，操作1为给第i天再分配ai个物品生产的任务，操作2为从第bi天开始修理这台机器（每次修理为独立的）。问你每次修理之后能完成多少生产任务。
很裸的线段树啊……单点更新和区间查询，注意区间查询的时候应该查询不属于query范围的区间。

#include <bits/stdc++.h> 
using namespace std;
typedef long long ll;
#define PB push_back
#define UB upper_bound
#define LB lower_bound
#define MP make_pair
#define pii pair<int,int>
#define vi vector<int>
#define vii vector<vi>
#define mst(x,y) memset(x,y,sizeof(x))
#define fr(x) freopen(x,"r",stdin)
#define fw(x) freopen(x,"w",stdout)
#define iin(x) scanf("%d",&x)
#define sp system("pause")
#define INF 1e9
#define MAXN 200005
#define PI acos(-1.0)
#define eps 1e-10
const ll MOD = 1e9 + 7;

int n,k,a,b,q,ans;

struct node
{
    int l,r,num,sum1,sum2;
}M[MAXN << 2];

void build(int ind,int l,int r)
{
    M[ind].l = l,M[ind].r = r;
    M[ind].num = 0,M[ind].sum1 = 0,M[ind].sum2 = 0;
    if(l != r)
        build(ind << 1,l,(l + r) >> 1),build(ind << 1 | 1,((l + r) >> 1) + 1,r);
}

void push_up(int ind)
{
    M[ind].sum1 = M[ind << 1].sum1 + M[ind << 1 | 1].sum1;
    M[ind].sum2 = M[ind << 1].sum2 + M[ind << 1 | 1].sum2;
}

void update(int ind,int pos,int value)
{
    if(M[ind].r < pos || M[ind].l > pos)
        return;
    if(M[ind].l == M[ind].r && M[ind].l == pos)
        M[ind].num += value,M[ind].sum1 = min(M[ind].num,a),M[ind].sum2 = min(M[ind].num,b);
    else
    {
        update(ind << 1,pos,value);
        update(ind << 1 | 1,pos,value);
        push_up(ind);
    }
}

void query(int ind,int l,int r)
{
    if(l <= M[ind].l && r >= M[ind].r)
        return;
    if(r < M[ind].l)
    {
        ans += M[ind].sum1;//cout << M[ind].l << " " << M[ind].r << " " << M[ind].sum2 << endl;sp;
        return;
    }
    else if(l > M[ind].r)
    {
        ans += M[ind].sum2;//cout << M[ind].l << " " << M[ind].r << " " << M[ind].sum1  << endl;sp;
        return;
    }
    else
    {
        query(ind << 1,l,r);
        query(ind << 1 | 1,l,r);
        push_up(ind);
    }

}

int main()   
{   

    cin >> n >> k >> a >> b >> q;
    build(1,1,n);
    for(int i = 0;i < q;i++)
    {
        int type,a,b;
        iin(type);
        if(type == 1)
        {
            iin(a),iin(b);
            update(1,a,b);
        }
        else
        {
            iin(a);
            ans = 0;
            query(1,a,a + k - 1);
            printf("%d\n",ans);
        }
    }
}

E. Package Delivery

有一条长d的路，一辆油箱上限为n的车，m个加油站，给出每个加油站的位置和油价，问你最少要付多少钱才能到达重点，起始油箱是满的。
贪心，对于当前一个位置的话，判断从此位置到终点为止的最低油价点有多远，如果在n的范围内，那么就加到刚好能到此位置的油，否则就加满游往下一个位置走。我们可以先用一个stack来处理出每个点到终点的最低油价点。
其实一开始我的想法不是这样的，我当初想的是对于每一个位置，我们判断离他最近的油价小于等于现在这个位置的油价的位置，如果能到达，就加油至刚好能到达位置，如果不能，就加满油，走到他所能走到的范围内最小的油价位置。结果无限WA7，然后和宝树讨论了一下，之后觉得如果不满足条件一的话直接加满后走到下一个加油站继续判断就可以了，结果就变成无限WA10了，想来想去并不懂为什么是错的啊……

#include <bits/stdc++.h> 
using namespace std;
typedef long long ll;
#define PB push_back
#define UB upper_bound
#define LB lower_bound
#define MP make_pair
#define pii pair<int,int>
#define vi vector<int>
#define vii vector<vi>
#define mst(x,y) memset(x,y,sizeof(x))
#define fr(x) freopen(x,"r",stdin)
#define fw(x) freopen(x,"w",stdout)
#define iin(x) scanf("%d",&x)
#define sp system("pause")
#define INF 1e9
#define MAXN 200005
#define PI acos(-1.0)
#define eps 1e-10
const ll MOD = 1e9 + 7;

int l[MAXN];
pii a[MAXN];
int main()   
{   
    int d,n,m;
    cin >> d >> n >> m;
    stack<int>st;
    for(int i = 1;i <= m;i++)
        iin(a[i].first),iin(a[i].second);
    a[0] = MP(0,0),a[m + 1] = MP(d,0);
    sort(a,a + m + 2);
    st.push(m + 1);l[m + 1] = m + 1;
    for(int i = m;i >= 0;i--)//处理出从第i个位置开始到末尾为止最低油价点的下标
    {
        while(a[st.top()].second > a[i].second)
            st.pop();
        l[i] = st.top();
        st.push(i);
    }
    int gas = n;
    ll sum = 0;
    for(int i = 1;i <= m + 1;i++)
    {
        gas -= a[i].first - a[i - 1].first;//依次跑到每个加油站
        if(gas < 0)
        {
            cout << -1 << endl;
            return 0;
        }
        int newgas = a[l[i]].first - a[i].first;//距离下一个最低点的距离
        int rem = min(newgas,n);//能否到达下一个最低点
        if(rem > gas)
            sum += (ll)(rem - gas) * a[i].second,gas = rem;
    }
    cout << sum << endl;
}
