A. Grandma Laura and Apples

初始有p个苹果，每次卖一半，如果苹果剩余个数是奇数，就把多出来的那个以一半的价钱顺便卖掉。现在给你每次卖的情况（单双），让你求出总共赚了多少钱。
水题……竟然因为longlong，wa了一次……

const ll MOD = 1e9 + 7;

int main()   
{

    cin >> n >> m;
    ll sum = 0,ss = 0;
    string s[111];
    for(int i = 0;i < n;i++)
    {
        cin >> s[i];

    }

B. Alice, Bob, Two Teams

给你一个AB串，并且给出每个位置的val，你可以进行一次前缀或者后缀串的翻转，要求操作后可能得到的最大价值（B价值和）。
先求个前缀和，然后枚举翻转位置就行了……

#include <bits/stdc++.h> 
using namespace std;
typedef long long ll;
#define PB push_back
#define UB upper_bound
#define LB lower_bound
#define MP make_pair
#define pii pair<int,int>
#define vi vector<int>
#define vii vector<vi>
#define mst(x,y) memset(x,y,sizeof(x))
#define fr(x) freopen(x,"r",stdin)
#define fw(x) freopen(x,"w",stdout)
#define iin(x) scanf("%d",&x)
#define sp system("pause")
#define INF 1e9
#define MAXN 500005
#define PI acos(-1.0)
#define eps 1e-10
const ll MOD = 1e9 + 7;

char s[MAXN];
int a[MAXN];
ll s1[MAXN],s2[MAXN];

int main()   
{   
    int n;
    cin >> n;
    for(int i = 0;i < n;i++)
        iin(a[i]);
    scanf("%s",s);
    ll maxn = 0;mst(s1,0);mst(s2,0);
    if(s[0] == 'A')
        s2[0] = a[0];
    else
        s1[0] = a[0];
    for(int i = 1;i < n;i++)
    {
        if(s[i] == 'A')
            s2[i] = s2[i - 1] + a[i],s1[i] = s1[i - 1];
        else
            s1[i] = s1[i - 1] + a[i],s2[i] = s2[i - 1];
    }
    maxn = max(s1[n - 1],s2[n - 1]);
    for(int i = 0;i < n - 1;i++)
    {
        maxn = max(maxn,max(s1[i] + s2[n - 1] - s2[i],s2[i] + s1[n - 1] - s1[i]));
       // cout << maxn << endl;
    }
    cout << maxn << endl;
}

C. The Smallest String Concatenation

给你n个字符串，让你把它们合成一个字符串，并且使其字典序最小。
写个cmp函数……

#include <bits/stdc++.h> 
using namespace std;
typedef long long ll;
#define PB push_back
#define UB upper_bound
#define LB lower_bound
#define MP make_pair
#define pii pair<int,int>
#define vi vector<int>
#define vii vector<vi>
#define mst(x,y) memset(x,y,sizeof(x))
#define fr(x) freopen(x,"r",stdin)
#define fw(x) freopen(x,"w",stdout)
#define iin(x) scanf("%d",&x)
#define sp system("pause")
#define INF 1e9
#define MAXN 500005
#define PI acos(-1.0)
#define eps 1e-10
const ll MOD = 1e9 + 7;

bool cmp(string s1,string s2)
{
    return s1 + s2 < s2 + s1;
}

int main()   
{   
    int n;
    iin(n);
    string ss[MAXN];
    char s[111];
    for(int i = 0;i < n;i++)
    {
        scanf("%s",s);
        ss[i] = (string)s;
    }   
    sort(ss,ss + n,cmp);
    for(int i = 0;i < n;i++)
    {
        printf("%s",ss[i].c_str());
    }
}

D. Longest Subsequence

给你一个数列，让你在[1,m]的范围内找出一个数p，使得数列中尽可能多的数的最小公倍数是p。
数据范围是1e6，所以不能太暴力……用类似素数筛的办法来解决。先过滤掉所有大于m的数，然后遍历一遍剩下的数，对于每一个数，用yin[n]来存储有多少个数是n的因子，枚举到n*k > m为止。最后从1枚举到m取个最大值就好了……

#include <bits/stdc++.h> 
using namespace std;
typedef long long ll;
#define PB push_back
#define UB upper_bound
#define LB lower_bound
#define MP make_pair
#define pii pair<int,int>
#define vi vector<int>
#define vii vector<vi>
#define mst(x,y) memset(x,y,sizeof(x))
#define fr(x) freopen(x,"r",stdin)
#define fw(x) freopen(x,"w",stdout)
#define iin(x) scanf("%d",&x)
#define sp system("pause")
#define INF 1e9
#define MAXN 500005
#define PI acos(-1.0)
#define eps 1e-10
const ll MOD = 1e9 + 7;

int n,m,a[1000005],b[1000005],yin[1000005];

int main()   
{   
    cin >> n >> m;
    for(int i = 0;i < n;i++)
    {
        iin(a[i]);
        if(a[i] <= m)
        b[a[i]]++;
    }
    for(int i = 1;i <= m;i++)
    {
        if(b[i])
        {
            for(int j = i;j <= m;j += i)
            {
                yin[j] += b[i];
            }
        }  
    }
    int minn = 0,sv = 1;
    for(int i = 1;i <= m;i++)
    {
        if(yin[i] > minn)
        {
            minn = yin[i];
            sv = i;
        }
    }
    cout << sv << " " << minn << endl;
    int ok = 0;
    for(int i = 0;i < n;i++)
    {
        if(sv % a[i] == 0)
        {
            if(ok)
                cout << " " << i + 1;
            else
                cout << i + 1,ok = 1;
        }
    }
}
