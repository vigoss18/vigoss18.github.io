题目链接：Twinkle Twinkle Little Star

今天模拟赛做到的一道题目，当时一看到的时候感觉自己做过，信心满满地开始做，结果……QAQ

题意是给你n个点，求用最小边长为多少的正方形能覆盖至少k个点。

因为点的范围在正负1e6内，所以不能直接上，经过预处理离散化之后，用dp处理出从原点到i,j位置总共有多少个点，用dp[i][j]表示。

模拟赛的时候我写到这里就无法写下去了，只会$n^4$的枚举。赛后仔细一想，才发现，因为是正方形，所以只要枚举一个点就够了也就是$n^2$，通过二分来枚举长度，判断这个长度内到底有没有能包含k个点的正方形，活到老学到老……

#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <algorithm>
#include <string>
#include <cstring>
#include <map>
#include <stack>
#include <vector>
#include <queue>
#include <set>
#include<functional>
#include<cctype>
#include<string.h>
#include<utility>
#define PB push_back
#define UB upper_bound
#define LB lower_bound
#define MP make_pair
#define pii pair<int,int>
#define vi vector<int>
#define vii vector<vi>
#define mst(x,y) memset(x,y,sizeof(x))
#define fr(x) freopen(x,"r",stdin)
#define fw(x) freopen(x,"w",stdout)
#define complete_unique(a) a.erase(unique(a.begin(),a.end()),a.end())
#define iin(x) scanf("%d",&x)
#define sp system("pause")
#define INF 1e9
#define MAXN 100005
#define PI acos(-1.0)
#define eps 1e-10
const int MOD = 1e9 + 7;
using namespace std;

int dp[1555][1555];
int svx[1555],svy[1555];
int mp[1555][1555];
int px[1555],py[1555];
int n,m;

vi r,c;
pii yuanben[1555],after[1555];

bool judge(int x)
{
    mst(svx,-1);mst(svy,-1);
    for(int i = 1;i < r.size();i++)
    {
        for(int j = i;j < r.size() && px[j] - px[i] <= x;j++)
        {
            svx[i] = j;
        }
    }

    for(int i = 1;i < c.size();i++)
    {
        for(int j = i;j < c.size() && py[j] - py[i] <= x;j++)
        {
            svy[i] = j;
        }
    }
    int temp;
    for(int i = 1;i < r.size() && svx[i] != -1;i++)
    {
        for(int j = 1;j < c.size() && svy[j] != -1;j++)
        {
            temp = dp[svx[i]][svy[j]] - dp[svx[i]][j - 1] - dp[i - 1][svy[j]] + dp[i - 1][j - 1];
            //cout << temp << endl;
            if(temp >= m)
                return true;
        }
    }
    return false;
}

int main()
{
    //freopen("1.txt","w",stdout);
    int cas = 1;
    while(cin >> n >> m)
    {
        mst(dp,0);mst(mp,0);mst(py,0);mst(px,0);
        r.clear(),c.clear();
        //cout << "ok";
        for(int i = 0;i < n;i++)
        {
            int a,b;
            iin(a),iin(b);
            yuanben[i] = MP(a,b);
            r.PB(a),c.PB(b);
        }

        r.PB(INF),r.PB(-INF),c.PB(INF),c.PB(-INF);
        sort(r.begin(),r.end());
        sort(c.begin(),c.end());
        complete_unique(r);
        complete_unique(c);

        for(int i = 0;i < n;i++)
        {
            int x = LB(r.begin(),r.end(),yuanben[i].first) - r.begin();
            int y = LB(c.begin(),c.end(),yuanben[i].second) - c.begin();
            px[x] = yuanben[i].first;
            py[y] = yuanben[i].second;
            mp[x][y]++;
        }

        dp[0][0] = mp[0][0];
        for(int i = 0;i < r.size();i++)
        {
            for(int j = 0;j < c.size();j++)
            {
                if(i > 0 && j > 0)
                {
                    dp[i][j] = max(dp[i][j],dp[i - 1][j] + dp[i][j - 1] - dp[i - 1][j - 1] + mp[i][j]);
                }
                else if(i > 0)
                {
                    dp[i][j] = max(dp[i][j],dp[i - 1][j] + mp[i][j]);
                }
                else if(j > 0)
                {
                    dp[i][j] = max(dp[i][j],dp[i][j - 1] + mp[i][j]);
                }
            }
        }
        //cout << px[2] - px[1] << endl;
        //cout << dp[2][2] << endl;
        int l = 0,r = INF;
        while(l < r)
        {
            int mid = (l + r) >> 1;
            if(judge(mid))
                r = mid;
            else
                l = mid + 1;
        }
        printf("Case %d: ",cas++);
        cout << (l + r) / 2 << endl;
    }
}
