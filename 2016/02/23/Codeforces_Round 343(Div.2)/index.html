A. Far Relative’s Birthday Cake

给你一个矩阵，求每行没列上有几对C的和。暴力

#include <bits/stdc++.h> 
using namespace std;
typedef long long ll;
#define PB push_back
#define UB upper_bound
#define LB lower_bound
#define MP make_pair
#define pii pair<int,int>
#define vi vector<int>
#define vii vector<vi>
#define mst(x,y) memset(x,y,sizeof(x))
#define fr(x) freopen(x,"r",stdin)
#define fw(x) freopen(x,"w",stdout)
#define sp system("pause")
#define INF 1e9
#define MAXN 100000
#define PI acos(-1.0)
#define eps 1e-10
const ll MOD = 1e9 + 7;

int main()
{
    int n;
    cin >> n;
    ll sum = 0;
    string s[111];
    for(int i = 0;i < n;i++)
        cin >> s[i];
    for(int i = 0;i < n;i++)
    {
        ll h = 0,hh = 0;
        for(int j = 0;j < n;j++){
            if(s[i][j] == 'C')
                h++;
            if(s[j][i] == 'C')
                hh++;
        }
        sum += hh * (hh - 1) + h * (h - 1);
    }
    cout << sum / 2;
}

B. Far Relative’s Problem

求给出一些线段，求最多有多少线段覆盖某一区间。注意的是有两种线段，必须都覆盖才行。
因为数据范围只有300多，所以存的时候hash一下，然后直接暴力遍历一遍就行了。

#include <bits/stdc++.h> 
using namespace std;
typedef long long ll;
#define PB push_back
#define UB upper_bound
#define LB lower_bound
#define MP make_pair
#define pii pair<int,int>
#define vi vector<int>
#define vii vector<vi>
#define mst(x,y) memset(x,y,sizeof(x))
#define fr(x) freopen(x,"r",stdin)
#define fw(x) freopen(x,"w",stdout)
#define sp system("pause")
#define INF 1e9
#define MAXN 100000
#define PI acos(-1.0)
#define eps 1e-10
const ll MOD = 1e9 + 7;

int main()
{
    int n;
    cin >> n;
    int m[444],maxn = 0;
    mst(m,0);
    for(int i = 0;i < n;i++)
    {
        char ch;
        int a,b;
        cin >> ch >> a >> b;
        if(ch == 'M')
        {
            for(int i = a;i <= b;i++)
                m[i] += 10000;
        }
        else
        {
            for(int i = a;i <= b;i++)
                m[i]++;
        }
    }
    for(int i = 1;i <= 366;i++)
    {
        maxn = max(maxn,min(m[i] % 10000,m[i] / 10000));
    }
    cout << maxn * 2;
}

C. Famil Door and Brackets

给你一个长度为n的字符串（包含左右括号），让你补全到m的长度，补充方式是原字符串为s，补充为q+s+p。合法的字符串有需要满足两个条件：1、左括号的总数必须等于右括号的总数。2、对于任意字符串的任意前缀，左括号的数量都必须大于等于右括号的数量。
Q神说可以推公式，组合数搞一搞但是我并不能想出来然后就让别人教了我一个dp的做法。dp[i][j]表示长度为i的字符串左括号比右括号多j个，预处理之后再for一遍就行了注意的是为了满足条件2，需要从q的长度为max(-mx,0)开始，mx代表s字符串中至少需要的左括号数。同时还要注意p的长度也有一定的限制条件。

#include <bits/stdc++.h> 
using namespace std;
typedef long long ll;
#define PB push_back
#define UB upper_bound
#define LB lower_bound
#define MP make_pair
#define pii pair<int,int>
#define vi vector<int>
#define vii vector<vi>
#define mst(x,y) memset(x,y,sizeof(x))
#define fr(x) freopen(x,"r",stdin)
#define fw(x) freopen(x,"w",stdout)
#define sp system("pause")
#define INF 1e9
#define MAXN 50005
#define PI acos(-1.0)
#define eps 1e-10
const int MOD = 1e9 + 7;

ll dp[2055][2055];//长度为i的串，(比)多j个的方案数

int main()
{
    int n,m;
    cin >> n >> m;
    string s;
    cin >> s;
    int x = 0,mx = INF;
    mst(dp,0);
    for(int i = 0;i < s.size();i++)
    {
        if(s[i] == '(')
            x++;
        else
            x--;
        mx = min(x,mx);
    }
    dp[0][0] = 1;
    for(int i = 1;i < 2005;i++)
    {
        for(int j = 0;j <= i;j++)
        {
            if(j)
                dp[i][j] = (dp[i - 1][j + 1] + dp[i - 1][j - 1]) % MOD;
            else
                dp[i][j] = dp[i - 1][j + 1];
        }
    }
    ll ans = 0;
    for(int i = max(-mx,0);i <= n - m;i++)
    {
        for(int j = max(-mx,0);j <= i;j++)
        {
            if(j + x > n - m - i)
                break;
            ans = (ans + dp[i][j] * dp[n - m - i][j + x]) % MOD;
        }
    }
    cout << ans << endl;
}

D. Babaei and Birthday Cake

简单地来说就是求最大上升子序列和，线段树、树状数组都可以做，我看到一种用map的办法。和求LIS那个nlogn的办法差不多，每次二分到最佳的插入位置，同时把对于当前来说已经不是最优解的记录全部删除。

#include <bits/stdc++.h> 
using namespace std;
typedef long long ll;
#define PB push_back
#define UB upper_bound
#define LB lower_bound
#define MP make_pair
#define pii pair<int,int>
#define vi vector<int>
#define vii vector<vi>
#define mst(x,y) memset(x,y,sizeof(x))
#define fr(x) freopen(x,"r",stdin)
#define fw(x) freopen(x,"w",stdout)
#define sp system("pause")
#define INF 1e18
#define MAXN 50005
#define PI acos(-1.0)
#define eps 1e-4

int main()
{
    map<ll,ll>mp;
    ll ans = 0;
    int n;
    cin >> n;
    mp[0] = 0;
    for(int i = 0;i < n;i++)
    {
        ll a,b;
        scanf("%lld%lld",&a,&b);
        ll v = a * a * b;
        auto it = mp.LB(v);
        it--;
        ll v2 = it -> second + v;
        mp[v] = v2;
        ans = max(ans,v2);
        it = mp.find(v);
        it++;
        while(it != mp.end())
        {
            if(it -> second <= v2)
            {
                auto it2 = it;
                it++;
                mp.erase(it2);
            }
            else
                break;
        }
    }
    printf("%.10f\n",ans * PI);
}
