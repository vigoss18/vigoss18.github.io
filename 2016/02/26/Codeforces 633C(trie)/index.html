Codeforces 663C(trie)

发表于 2016-02-26 | 分类于 ACM-数据结构 |

希望以后每次CF的C都能做出来（手动斜眼）

题意是给你一个字符串，这个字符串曾经经过三种处理：1、所有的字母都替换为小写字母。2、把所有的单词翻转。3、去掉所有的空格。然后给你m个子串。让你从这些子串中选出一些，还原出原始串。

我的解法是把所有的子串建一个trie树，然后直接dfs一下……一开始reverse大字符串T了，换成reverse所有的子串就过了……不明觉厉

#include <bits/stdc++.h> 
using namespace std;
typedef long long ll;
#define PB push_back
#define UB upper_bound
#define LB lower_bound
#define MP make_pair
#define pii pair<int,int>
#define vi vector<int>
#define vii vector<vi>
#define mst(x,y) memset(x,y,sizeof(x))
#define fr(x) freopen(x,"r",stdin)
#define fw(x) freopen(x,"w",stdout)
#define sp system("pause")
#define INF 1e9
#define MAXN 1000005
#define PI acos(-1.0)
#define eps 1e-10
const ll MOD = 1e9 + 7;

int n,m,cnt = 1;
char s[11111];
string ns,sv[100005];

struct Tire
{
    int nxt[26],flag;
    void init()
    {
        mst(nxt,-1);
        flag = -1;
    }
}trie[MAXN];

void init()
{
    cnt = 1;
    trie[1].init();
}

void Insert(string s,int num)
{
    int p = 1;
    for(int i = 0;i < s.size();i++)
    {
        int pos = s[i] - 'a';
        if(trie[p].nxt[pos] == -1)
        {
            trie[p].nxt[pos] = ++cnt;
            trie[cnt].init();
        }
        p = trie[p].nxt[pos];
    }
    trie[p].flag = num;
}

int ou[10005],oucnt = 0,ok = 0,nowpos = 0;

void dfs(int id)
{
    if((nowpos == n && id == 1) || ok)
    {
        ok = 1;
        return;
    }
    int npos = ns[nowpos++] - 'a';
    if(trie[id].nxt[npos] > 0)
    {
        if(trie[trie[id].nxt[npos]].flag > 0)
        {
            ou[oucnt++] = trie[trie[id].nxt[npos]].flag;
            dfs(1);
            if(ok)
                return;
            oucnt--;
            nowpos--;
        }
        dfs(trie[id].nxt[npos]);
        if(ok)
            return;
        nowpos--;
    }
}

int main()   
{   
    init();
    scanf("%d",&n);
    scanf("%s",s);
    ns = (string)s;mst(ou,0);
    //reverse(ns.begin(),ns.end());
    scanf("%d",&m);
    for(int i = 1;i <= m;i++)
    {
        string trans;
        char nsv[1111];
        scanf("%s",nsv);
        sv[i] = (string)nsv;
        trans = sv[i];
        reverse(trans.begin(),trans.end());
        for(int j = 0;j < trans.size();j++)
        {
            if(trans[j] < 'a')
                trans[j] += 32;
        }
        Insert(trans,i);
    }
    dfs(1);
    for(int i = 0;i < oucnt;i++)
    {
        if(!i)
        {
            printf("%s",sv[ou[i]].c_str());
        }
        else
            printf(" %s",sv[ou[i]].c_str());
    }
    cout << endl;
}
