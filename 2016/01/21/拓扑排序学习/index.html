拓扑排序学习

发表于 2016-01-21 | 分类于 ACM |

今天模拟赛做到一道拓扑排序的题，前两天还因为不知道拓扑排序，在数据结构考试里扣了两分……
于是我今天痛下决心学一下……（好叭，痛什么决心……这是个基本算法）

拓扑排序是应用在有向无环图里面的一种排序。简单来说就是先把所有入度为0的点找到，删掉所有与他相连的边，然后再找入度为0的点，直到找不到为止。顶点的删除顺序就是拓扑排序。

HDU4857

题目链接: 逃生

中文题意就不解释了。
因为题目要求的是先满足1在尽可能前面，而正向的拓扑排序，明显不会满足这个条件。于是我们用逆向思维……把图反过来，每次把排在最后的人扣出来，即尽可能得使没钱的人排在后面，这样形成的拓扑排序反一下就是答案了……

typedef long long ll;
#define PB push_back

#define fw(x) freopen(x,"w",stdout)
#define INF 1e17
#define MAXN 30005
#define PI acos(-1.0)
#define eps 1e-10


vi Edge[MAXN];
int in[MAXN],n,m,gg[MAXN],cnt = 0;

void fuck_you()
{
    priority_queue<int> q;
    for(int i = 1;i <= n;i++)
        if(!in[i])
            q.push(i);
    int ok = 0;
    while(!q.empty())
    {
        int h = q.top();
        gg[cnt++] = h;
        q.pop();
        for(int i = 0;i < Edge[h].size();i++)
        {
            in[Edge[h][i]]--;
            if(!in[Edge[h][i]])
                q.push(Edge[h][i]);
        }
    }
}

int main()
{
    int t;
    cin >> t;
    while(t--)
    {

        cin >> n >> m;
        mst(in,0);mst(gg,0);
        cnt = 0;
        for(int i = 1;i <= n;i++)
            Edge[i].clear();
        for(int i = 0;i < m;i++)
        {
            int a,b;
            scanf("%d %d",&a,&b);
            Edge[b].PB(a);
            in[a]++;
        }
        fuck_you();
        for(int i = n - 1;i >= 1;i--)
            cout << gg[i] << " ";
        cout << gg[0] << endl;
    }

ZOJ2193

题目链接: Window Pains

题意是给你如图所示的一块板，填充方法固定，但是填充顺序不固定，问给你的终态板是否合法。
我们可以这么想，一块本该覆盖1的区域有了其他数字，那么一定是先放的1，其余数字都是后放的，遍历所有九个块，就可以得出填充关系的DAG，然后拓扑排序判断一下是否存在环即可（即队列清空后还有数字没有进过队列），存在环就是不合法的。

#include <bits/stdc++.h> 
using namespace std;
typedef long long ll;
#define PB push_back
#define UB upper_bound
#define LB lower_bound
#define MP make_pair
#define pii pair<int,int>
#define vi vector<int>
#define vii vector<vi>
#define mst(x,y) memset(x,y,sizeof(x))
#define fr(x) freopen(x,"r",stdin)
#define fw(x) freopen(x,"w",stdout)
#define INF 1e17
#define MAXN 30005
#define PI acos(-1.0)
#define eps 1e-10

vi Edge[11];
int in[11];

void solve()
{
    queue<int>q;
    int h = 0;
    for(int i = 1;i < 10;i++)
        if(!in[i])
            q.push(i);
    while(!q.empty())
    {
        int po = q.front();
        q.pop();h++;
        for(int i = 0;i < Edge[po].size();i++)
        {
            in[Edge[po][i]]--;
            if(!in[Edge[po][i]])
                q.push(Edge[po][i]);
        }
    }
    if(h == 9)
        cout << "THESE WINDOWS ARE CLEAN\n";
    else
        cout << "THESE WINDOWS ARE BROKEN\n";
}

int main()
{
    string s;
    while(cin >> s && s == "START")
    {
        mst(in,0);
        int mp[11][11];
        for(int i = 0;i < 11;i++)
            Edge[i].clear();
        for(int i = 0;i < 4;i++)
        {
            for(int j = 0;j < 4;j++)
            {
                cin >> mp[i][j];
            }
        }
        cin >> s;
        int st = 1;
        for(int i = 0;i < 3;i++)
        {
            for(int j = 0;j < 3;j++)
            {
                if(mp[i][j] != st)
                    Edge[st].PB(mp[i][j]),in[mp[i][j]]++;
                if(mp[i][j + 1] != st)
                    Edge[st].PB(mp[i][j + 1]),in[mp[i][j + 1]]++;
                if(mp[i + 1][j] != st)
                    Edge[st].PB(mp[i + 1][j]),in[mp[i + 1][j]]++;
                if(mp[i + 1][j + 1] != st)
                    Edge[st].PB(mp[i + 1][j + 1]),in[mp[i + 1][j + 1]]++;
                st++;
            }
        }
        solve();
    }
}

HDOJ1811

题目链接: Rank of Tetris

中文题，也是基本的拓扑排序，不过加上了并查集。通过并查集把所有rating相等的点缩成一个点之后，在此基础上进行拓扑排序。如果某一时间内队列里存在多余一个点的情况，说明拓扑排序不唯一，如果排序个数小于缩点后的总个数，则说明存在冲突。

不得不说我几乎每次写并查集都会犯一个sb错误，就是把pre[i]当成find(i)来用，今晚wa了半天结果却是并查集的锅，桑心。

#include <bits/stdc++.h> 
using namespace std;
typedef long long ll;
#define PB push_back
#define UB upper_bound
#define LB lower_bound
#define MP make_pair
#define pii pair<int,int>
#define vi vector<int>
#define vii vector<vi>
#define mst(x,y) memset(x,y,sizeof(x))
#define fr(x) freopen(x,"r",stdin)
#define fw(x) freopen(x,"w",stdout)
#define INF 1e17
#define MAXN 20005
#define PI acos(-1.0)
#define eps 1e-10

int n,m,in[MAXN],pre[MAXN],flag0,sum;
vi Edge[MAXN];

void init()
{
    for(int i = 0;i < MAXN;i++)
        pre[i] = i;
}

int find(int x)
{
    return (x == pre[x] ? x : pre[x] = find(pre[x]));
}

void mix(int x,int y)
{
    int rx = find(x),ry = find(y);
    if(rx != ry)
        pre[rx] = ry,sum--;
}

void solve()
{
    queue<int> q;
    int ok1 = 0,ok2 = 0;
    for(int i = 0;i < n;i++)
        if(!in[find(i)] && find(i) == i)
            q.push(find(i));
    while(!q.empty())
    {
        int h = q.front();
        q.pop();
        ok2++;
        if(!q.empty())
        {
            ok1 = 1;
        }
        for(int i = 0;i < Edge[h].size();i++)
        {
            in[Edge[h][i]]--;
            if(!in[Edge[h][i]])
                q.push(Edge[h][i]);
        }
    }
    if(ok2 != sum || flag0)
        cout << "CONFLICT\n";
    else if(ok1)
        cout << "UNCERTAIN\n";
    else 
        cout << "OK\n";
}

int main()
{
    while(cin >> n >> m)
    {
        init();
        flag0 = 0;
        sum = n;
        int a[MAXN],b[MAXN],flag[MAXN];
        mst(flag,0);
        for(int i = 0;i < MAXN;i++)
            Edge[i].clear();
        mst(in,0);
        char c;
        for(int i = 0;i < m;i++)
        {
            cin >> a[i] >> c >> b[i];
            if(c == '=')
                mix(a[i],b[i]);
            else if(c == '>')
                flag[i] = 1;
            else
                flag[i] = -1;
        }
        for(int i = 0;i < m;i++)
        {
            if(flag[i] == 1)
                Edge[find(a[i])].PB(find(b[i])),in[find(b[i])]++;
            else if(flag[i] == -1)
                Edge[find(b[i])].PB(find(a[i])),in[find(a[i])]++;
            if(flag[i] && find(b[i]) == find(a[i]))
                flag0 = 1;
        }
        solve();
    }
}
