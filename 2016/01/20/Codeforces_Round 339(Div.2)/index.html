A. Link/Cut Tree

输入a,b,c，按递增输出所有处于a到b之间的属于$c^n$的数。
需要注意的点是如果你每次判断的是t <= b的话，最后一次之后会出现越界的情况，然后你就wa了……
所以判断的时候用b / c < t来判断……

#include <bits/stdc++.h> 
using namespace std;
typedef long long ll;
#define PB push_back
#define UB upper_bound
#define LB lower_bound
#define MP make_pair
#define pii pair<int,int>
#define vi vector<int>
#define vii vector<vi>
#define mst(x) memset(x,0,sizeof(x))
#define fr(x) freopen(x,'r',stdin)
#define fw(x) freopen(x,'w',stdout)

int main()
{
    int num=0;
    ll l,r,k;
    scanf("%I64d%I64d%I64d",&l,&r,&k);
    for(long long i=1;i<=r;i*=k)
    {
        if(i>=l&&i<=r)
        {
            num++;
            printf("%I64d ",i);
        }
        if(r/i<k) 
            break;
    }
    if(num==0) 
        printf("-1");
    return 0;
}

B. Gena’s Code

给你一个长度为n的序列，其中至多只有一个“不美丽数”，美丽数就是只存在一个1的数，问你所有数字的乘积是多少。
如果一个数是美丽数，就记录length - 1，否则记录这个数是多少，输出的时候先输出不美丽数，然后输出记录的长度个的0，注意如果没有不美丽数需要先输出1，所以先把记录的数置为1就好了

#include <bits/stdc++.h> 
using namespace std;
typedef long long ll;
#define PB push_back
#define UB upper_bound
#define LB lower_bound
#define MP make_pair
#define pii pair<int,int>
#define vi vector<int>
#define vii vector<vi>
#define mst(x) memset(x,0,sizeof(x))
#define fr(x) freopen(x,'r',stdin)
#define fw(x) freopen(x,'w',stdout)

int main()
{
    int n,ok = 1,h = 0;
    string s,st = "1";
    cin >> n;
    for(int i = 0;i < n;i++)
    {
        cin >> s;
        int ct = 0;
        for(int j = 0;j < s.length();j++)
            if(s[j] == '0')
                ct++;
        if(s == "0")
            ok = 0;  
        else if(ct + 1 == s.length() && s[0] == '1')
            h += ct;
        else
            st = s;
        //cout << h << endl;
    }
    if(!ok)
        cout << 0 << endl;
    else
    {
        cout << st;
        for(int i = 0;i < h;i++)
            cout << 0;
    }
}

C. Peter and Snow Blower

几何题，给你一个多边形，让他以一个点为圆心，做圆周运动，问你经过的面积有多少。只要求这个多边形离圆心最近和最远的距离就好了。
需要注意的是最近的距离可能不是两点之间的距离，而是以多边形一条边为底所做的高（前提是这个高在形成的三角形内）
我再写的时候先以为只要是钝角三角形就一定不满足，后来发现只要多边形的边是钝角三角形最长的那条边，也一样满足高在三角形内，其次就是我inf开到1e9太小了，为了减小精度误差我并没有使用sqrt，而是直接在平方的基础上进行运算，于是1e9就hold不住了……雪崩调试了一个多小时。

#define eps 1e-10

int t;
  double px,py,x[100005],y[100005],minn = INF,maxx = -INF;

  double getmin(int num)
{
     double h[3],h1[3],sv;
    h[0] = (px - x[num]) * (px - x[num]) + (py - y[num]) * (py - y[num]);
    h[2] = (x[num] - x[(num + 1) % t]) * (x[num] - x[(num + 1) % t]) + (y[num] - y[(num + 1) % t]) * (y[num] - y[(num + 1) % t]);
    h[1] = (px - x[(num + 1) % t]) * (px - x[(num + 1) % t]) + (py - y[(num + 1) % t]) * (py - y[(num + 1) % t]);
    for(int i = 0;i < 3;i++)
        h1[i] = h[i];
    //sort(h1,h1 + 3);
    //cout << h[0] << " " << h[1] << " " << h[2] << endl;
    if(h1[0] + h1[2] < h1[1] || h1[1] + h1[2] < h1[0])
        return INF;
    return h[0] - (h[2] - h[1] + h[0]) * (h[2] - h[1] + h[0]) / 4/ h[2];
}

D. Skills

n个科目，每个科目最高分为A，现在最多能提高m分。最后权值是两种分数算法之和，第一种算法是达到A分的科目个数cf，第二种算法是最低分cm。求提高单科分数之后，最后的权值最大是多少。并打印出方案。

思路就是贪心有几门课达到了A分，然后二分最低分是几分，里面套一个二分来算需要多少分数……wa了好久……

#include <bits/stdc++.h> 
using namespace std;
typedef long long ll;
#define PB push_back
#define UB upper_bound
#define LB lower_bound
#define MP make_pair
#define pii pair<int,int>
#define vi vector<int>
#define vii vector<vi>
#define mst(x,y) memset(x,y,sizeof(x))
#define fr(x) freopen(x,"r",stdin)
#define fw(x) freopen(x,"w",stdout)
#define sp system("pause")
#define INF 1e17
#define MAXN 100011
#define PI acos(-1.0)
#define eps 1e-10

bool cmp(pii a,pii b)
{
    return a.first < b.first;
}

bool cmpp(pii a,pii b)
{
    return a.second < b.second;
}

int main()
{
    ll n,a,cf,cm,m,sum[100005] = {0},mmid,ccnt,flag2;
    int pos;
    ll maxx = -1,flag = 0;
    pii f[100005];
    cin >> n >> a >> cf >> cm >> m;
    for(int i = 0;i < n;i++)
    {
        cin >> f[i].first;
        f[i].second = i;
    }
    sort(f,f + n,cmp);
    sum[0] = (ll)f[0].first;
    for(int i = 1;i < n;i++)
    {
        sum[i] = sum[i - 1] + (ll)f[i].first;
    }
    if(a * n - sum[n] <= m)
    {
        cout << n * cf + a * cm << endl;
        cout << a;
        for(int i = 1;i < n;i++)
            cout << " " << a;
        return 0;
    }
    for(int i = n;i >= 0;i--)
    {
        ll lef = m - (a * (n - i) - sum[n - 1] + (i == 0 ? 0 : sum[i - 1]));
        if(lef < 0)
            break;
        ll l = 0,r = a;
        while(l <= r)
        {
            ll mid = (l + r) >> 1;
            pii mids;
            mids.first = mid;
            pos = LB(f,f + i,mids) - f;
            if(mid * pos - sum[pos - 1] > lef)
                r = mid - 1;
            else
                l = mid + 1;
        }
        if((n - i) * cf + ((l + r) >> 1) * cm > maxx)
        {
            maxx = (n - i) * cf + ((l + r) >> 1) * cm;
            mmid = (l + r) >> 1;
            flag = i;flag2 = pos;
        }
    }
    for(int i = n - 1;i >= flag;i--)
        f[i].first = a;
    for(int i = 0;i <= flag2 - 1;i++)
            f[i].first = mmid;
    sort(f,f + n,cmpp);
    cout << maxx << endl;
    cout << f[0].first;
    for(int i = 1;i < n;i++)
        cout << " " << f[i].first;
}
