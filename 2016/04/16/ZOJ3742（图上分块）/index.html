ZOJ3742（图上分块）

发表于 2016-04-16 | 分类于 ACM |

昨天模拟赛的时候做到一道以前做过的题……只是数据比以前大了……之前是用暴力过的，结果这根本不敢写暴力，结果学长事后告诉我他们就是用暴力过的……而且比我学正确做法过的时间还要短……无语。

题意就是给你一个无向图，n个点，m条边，有一些操作：Q + 查询正边权值和 Q - 查询负边权值和 Q A 查询所有边权值和 C X 把所有与X邻接的边的权值取反。

做法是把顶点度数大于sqrt(n)的点作为高位点，其余的点作为低位点，低位点之间可以直接暴力处理，高位点之间也可以直接暴力处理，低位点和高位点，通过flag标记，每操作点X一次，就把标记更改一下，下次低位点遇到这个高位点的时候，就修改一下值就行了……

#include <bits/stdc++.h> 
using namespace std;
typedef long long ll;
#define PB push_back
#define UB upper_bound
#define LB lower_bound
#define MP make_pair
#define pii pair<int,int>
#define vi vector<int>
#define vii vector<vi>
#define mst(x,y) memset(x,y,sizeof(x))
#define fr(x) freopen(x,"r",stdin)
#define fw(x) freopen(x,"w",stdout)
#define sp system("pause")
#define INF 1e18
#define MAXN 50005
#define hash hhash
#define PI acos(-1.0)
#define eps 1e-4

struct edge
{
    int u,v;
    ll w;
}edg[50005];

int vis[50005],flag[50005],cnt,hash[50005];
ll sum1[555][555][2],sum2[555][2],allp,alld;
int n,m,q;
vector<pair<int,ll> > v[50005];

void pre()
{
    int lim = sqrt(n + 0.5);
    for(int i = 1;i <= n;i++)
    {
        if(vis[i] > lim)
            hash[i] = ++cnt;
    }
    for(int i = 0;i < m;i++)
    {
        if(edg[i].w > 0) allp += edg[i].w;
        else alld += edg[i].w;
       // cout << allp << " " << alld << endl;
        int u = edg[i].u,v = edg[i].v;
        if(hash[u] && hash[v])
        {
            if(edg[i].w > 0)
            {
                sum1[hash[u]][hash[v]][0] += edg[i].w;
                sum1[hash[v]][hash[u]][0] += edg[i].w;
            }
            else
            {
                sum1[hash[u]][hash[v]][1] += edg[i].w;
                sum1[hash[v]][hash[u]][1] += edg[i].w;
            }
        }
        if(hash[u]) sum2[hash[u]][edg[i].w < 0] += edg[i].w;
        if(hash[v]) sum2[hash[v]][edg[i].w < 0] += edg[i].w;
    }
}

void change(int x)
{
    if(hash[x])
    {
        allp += -sum2[hash[x]][1];
        allp -= sum2[hash[x]][0];
        alld += -sum2[hash[x]][0];
        alld -= sum2[hash[x]][1];
        swap(sum2[hash[x]][1],sum2[hash[x]][0]);
        sum2[hash[x]][0] *= -1,sum2[hash[x]][1] *= -1;

        //cout << allp << " " <<  alld << endl;
        for(int i = 1;i <= cnt;i++)
        {
            if(i == hash[x]) continue;
            sum2[i][0] -= sum1[i][hash[x]][0];
            sum2[i][0] += -sum1[i][hash[x]][1];
            sum2[i][1] -= sum1[i][hash[x]][1];
            sum2[i][1] += -sum1[i][hash[x]][0];
            swap(sum1[i][hash[x]][0],sum1[i][hash[x]][0]);
            sum1[i][hash[x]][0] *= -1,sum1[i][hash[x]][1] *= -1;
            swap(sum1[hash[x]][i][0],sum1[hash[x]][i][0]);
            sum1[hash[x]][i][0] *= -1,sum1[hash[x]][i][1] *= -1;
        }
        flag[x] ^= 1;
    }
    else
    {
        flag[x] ^= 1;int tmp;
        for(int i = 0;i < v[x].size();i++)
        {
            if(flag[x] ^ flag[v[x][i].first]) tmp = -v[x][i].second;
            else tmp = v[x][i].second;
            if(tmp > 0)
            {
                allp += tmp;
                alld -= -tmp;
                if(hash[v[x][i].first])
                {
                    sum2[hash[v[x][i].first]][0] += tmp;
                    sum2[hash[v[x][i].first]][1] -= -tmp;
                }
            }
            else
            {
                allp -= -tmp;
                alld += tmp;
                if(hash[v[x][i].first])
                {
                    sum2[hash[v[x][i].first]][0] -= -tmp;
                    sum2[hash[v[x][i].first]][1] += tmp;
                }
            }

        }
    }
}

int main()
{
    int ok = 0;
    while(cin >> n >> m >> q)
    {
        if(ok) cout << endl;
        else ok = 1;
        for(int i = 0;i < 50005;i++) v[i].clear();
        mst(vis,0);mst(sum1,0);mst(sum2,0);mst(flag,0);mst(hash,0);
        allp = 0,alld = 0,cnt = 0;
        for(int i = 0;i < m;i++)
        {
            scanf("%d %d %lld",&edg[i].u,&edg[i].v,&edg[i].w);
            vis[edg[i].u]++;vis[edg[i].v]++;
            v[edg[i].u].PB(MP(edg[i].v,edg[i].w));
            v[edg[i].v].PB(MP(edg[i].u,edg[i].w));
        }
        pre();getchar();
        //cout << allp << " " << alld << endl;
        for(int i = 0;i < q;i++)
        {
            char ch,ch1;
            int p;
            cin >> ch;
            if(ch == 'Q')
            {
                cin >> ch1;
                //cout << "sasasas" << ch1 << endl;
                if(ch1 == 'A')
                    printf("%lld\n",allp + alld);
                else if(ch1 == '+')
                    printf("%lld\n",allp);
                else
                    printf("%lld\n",alld);
            }
            else
            {
                scanf("%d",&p);
                change(p);
            }
        }
    }
}
