CSUOJ1711（模拟）

发表于 2016-04-17 | 分类于 ACM |

题目链接：http://acm.csu.edu.cn/OnlineJudge/problem.php?id=1711

题意就是给你一个二叉树，其实代表的就是一个族谱，问你点x和点y的关系

做法是首先对于两个点求他们的LCA，如果LCA离高位点差一个单位距离，那么他们的关系就是兄弟姐妹关系，如果差两个就是”1st cousin”关系，差三个就是”2nd cousin”关系，差四个就是”3rd cousin”关系。然后需要判断这两个点差几层，差一层就是（侄子）儿女关系，差两层就要加个”grand”前缀，差三层就要再加一个”great-“前缀，四层同理。当然一开始就要判断是否输出”kin”，如果你的LCA >= 2的话，在(min(cot1,cot2) > 4 || abs(cot1 - cot2) >= 4)时输出”kin”，否则在abs(cot1 - cot2) > 4时输出”kin”，在读懂题之后就是水题了……

#include <cstdio>  
#include <cstring>  
#include <algorithm>  
#include<iostream>
#include<queue>
#include<string>
#include<cstdlib>
#include<cmath>
using namespace std;  
const int maxn = 57;  
#define sp system("pause")
#define MAXN 100005
#define eps 1e-4
#define complete_unique(x) x.erase(unique(x.begin(),x.end()),x.end())
#define distance ddddd
#define ll long long

char sex[35000];

int main()
{
    int n,m;
    char ch;
    while(cin >> n >> m>>ch && n != -1 && m != -1)
    {
        int cot1=0,cot2=0,x=n,y=m;
        while(1)
        {
            if(abs(x-y)>0)
            {
                if(x>y)
                {
                    x=(x-1)>>1;
                    cot1++;
                }
                else
                {
                    y=(y-1)>>1;
                    cot2++;
                }
            }
            else if(x==y)break;
        }
        //cout<<cot1<<cot2;
        if(min(cot1,cot2)>=2)
        {
            if(abs(cot1-cot2)>=4||min(cot1,cot2)>4)
            {
                printf("kin\n");
                continue;
            }
        }
        else
        {
            if(abs(cot1-cot2)>4)
            {
                printf("kin\n");
                continue;
            }
        }
        if(cot1==0&&cot2==0)
            printf("self\n");
        else
        {
            if(cot2>cot1)
            {
                if(cot1==1)
                {
                    if(cot1==cot2)
                        printf("%s\n",(ch=='F'?"sister":"brother"));
                    else
                    {
                        for(int i=0;i<cot2-cot1-2;i++)
                            printf("great-");
                        if(cot2-cot1>=2)printf("grand");
                        printf("%s\n",(ch=='M'?"nephew":"niece"));
                    }
                }
                else if(cot1==0)
                {
                    for(int i=0;i<cot2-cot1-2;i++)
                        printf("great-");
                    if(cot2-cot1>=2)printf("grand");
                    printf("%s\n",(ch=='M'?"son":"daughter"));
                }
                else if(cot1==2)
                {
                    printf("1st cousin");
                    if(cot1==cot2)
                        printf("\n");
                    else if(cot2-cot1==1)
                        printf(" once removed\n");
                    else if(cot2-cot1==2)
                        printf(" twice removed\n");
                    else if(cot2-cot1==3)
                        printf(" thrice removed\n");
                }
                else if(cot1==3)
                {
                    printf("2nd cousin");
                    if(cot1==cot2)
                        printf("\n");
                    else if(cot2-cot1==1)
                        printf(" once removed\n");
                    else if(cot2-cot1==2)
                        printf(" twice removed\n");
                    else if(cot2-cot1==3)
                        printf(" thrice removed\n");
                }
                else if(cot1==4)
                {
                    printf("3rd cousin");
                    if(cot1==cot2)
                        printf("\n");
                    else if(cot2-cot1==1)
                        printf(" once removed\n");
                    else if(cot2-cot1==2)
                        printf(" twice removed\n");
                    else if(cot2-cot1==3)
                        printf(" thrice removed\n");
                }
            }
            else
            {

                if(cot2==1)
                {
                    if(cot1==cot2)
                        printf("%s\n",(ch=='F'?"sister":"brother"));
                    else
                    {
                        for(int i=0;i<cot1-cot2-2;i++)
                            printf("great-");
                        if(cot1-cot2>=2)printf("grand");
                        printf("%s\n",(ch=='M'?"uncle":"aunt"));
                    }
                }
                else if(cot2==0)
                {
                    for(int i=0;i<cot1-cot2-2;i++)
                        printf("great-");
                    if(cot1-cot2>=2)printf("grand");
                    printf("%s\n",(ch=='M'?"father":"mother"));
                }
                else if(cot2==2)
                {
                    printf("1st cousin");
                    if(cot1==cot2)
                        printf("\n");
                    else if(cot1-cot2==1)
                        printf(" once removed\n");
                    else if(cot1-cot2==2)
                        printf(" twice removed\n");
                    else if(cot1-cot2==3)
                        printf(" thrice removed\n");
                }
                else if(cot2==3)
                {
                    printf("2nd cousin");
                    if(cot1==cot2)
                        printf("\n");
                    else if(cot1-cot2==1)
                        printf(" once removed\n");
                    else if(cot1-cot2==2)
                        printf(" twice removed\n");
                    else if(cot1-cot2==3)
                        printf(" thrice removed\n");
                }
                else if(cot2==4)
                {
                    printf("3rd cousin");
                    if(cot1==cot2)
                        printf("\n");
                    else if(cot1-cot2==1)
                        printf(" once removed\n");
                    else if(cot1-cot2==2)
                        printf(" twice removed\n");
                    else if(cot1-cot2==3)
                        printf(" thrice removed\n");
                }
            }
        }
    }
    return 0;
}
