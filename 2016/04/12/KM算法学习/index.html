KM算法学习

发表于 2016-04-12 | 分类于 ACM |

啊……今天学着套了一下KM的模板。

KM算法求的是完备匹配下的最大权匹配：在一个二分图里，左顶点为x，右顶点为y，对于每组左右连接的点有权值w，求一种匹配使得所有的w之和最大。

KM算法是建立在hungary算法基础上的一种算法，因为离省赛有点近了，也就不仔细研究代码的细节了，先让自己会用。扔个模板先：

#include<string>
#include<iostream>
#include<cstring>
#include<algorithm>
#include<cstdio>
#include<map>
#include<vector>
using namespace std;

const int N = 310;
const int INF = 0x3f3f3f3f;
int nx,ny;//两边点数
int g[555][555];//二分图描述    
int linker[N],lx[N],ly[N];//y中各点匹配状态，xy中的点标号
int slack[N];
bool visx[N],visy[N];

bool DFS(int x)
{
    visx[x] = true;
    for(int y = 0;y < ny;y++)
    {
        if(visy[y]) continue;
        int tmp = lx[x] + ly[y] - g[x][y];
        if(tmp == 0)
        {
            visy[y] = true;
            if(linker[y] == - 1 || DFS(linker[y]))
            {
                linker[y] = x;
                return true;
            }
        }
        else if(slack[y] > tmp)
            slack[y] = tmp;
    }
    return false;
}

int KM()
{
    memset(linker,-1,sizeof(linker));
    memset(ly,0,sizeof(ly));
    for(int i = 0;i < nx;i++)
    {
        lx[i] = -INF;
        for(int j = 0;j < ny;j++)
            if(g[i][j] > lx[i])
                lx[i] = g[i][j];
    }
    for(int x = 0;x < nx;x++)
    {
        for(int i = 0;i < ny;i++)
            slack[i] = INF;
        while(1)
        {
            memset(visx,false,sizeof(visx));
            memset(visy,false,sizeof(visy));
            if(DFS(x)) break;
            int d = INF;
            for(int i = 0;i < ny;i++)
                if(!visy[i] && d > slack[i])
                    d = slack[i];
            for(int i = 0;i < nx;i++)
                if(visx[i])
                    lx[i] -= d;
            for(int i = 0;i < ny;i++)
            {
                if(visy[i]) ly[i] += d;
                else slack[i] -= d;
            }
        }
    }
    int res = 0;
    for(int i = 0;i < ny;i++)
    {
        if(linker[i] != -1)
            res += g[linker[i]][i];
    }
    return res;
}

int main()
{
    int n;
    while(cin >> n)
    {
        for(int i = 0;i < n;i++)
        {
            for(int j = 0;j < n;j++)
            {
                scanf("%d",&g[i][j]);
            }
        }
        nx = ny = n;
        cout << KM() << endl;
    }
}

HDU1533

题目链接:http://acm.hdu.edu.cn/showproblem.php?pid=1533

给你一个图，要求每个人都要进一个房子，要求最后总的哈密顿距离最短，这里要求的是最短，所以需要稍微修改一下，在读入的时候把所有的距离处理成负值，最后输出的时候再处理一下就好了……

#include<string>
#include<iostream>
#include<cstring>
#include<algorithm>
#include<cstdio>
#include<map>
#include<vector>
using namespace std;

const int N = 310;

int main()
{
    int n,m;
    while(cin >> n >> m && n + m != 0)
    {
        vector<pair<int,int>>h,mm;
        for(int i = 0;i < n;i++)
        {
            string s;
            cin >> s;
            for(int j = 0;j < m;j++)
            {
                if(s[j] == 'H')
                    h.push_back(make_pair(i,j));
                if(s[j] == 'm')
                    mm.push_back(make_pair(i,j));
            }
        }
        for(int i = 0;i < mm.size();i++)
        {
            for(int j = 0;j < h.size();j++)
            {
                g[i][j] = -abs(mm[i].first - h[j].first) - abs(mm[i].second - h[j].second);
            }
        }
        nx = mm.size(),ny = h.size();
        cout << -KM() << endl;
    }
}
